149 void US_2dPlot::calculate()
150 {
151 
152  x .resize( resolution );
153  y .resize( resolution );
154  double inc = 1.0/resolution;
155 
156  for ( int i = 0; i < resolution; i++ )
157  {
158  x[ i ] = i * inc;
159  }
160  switch (model)
161  {
162  case 1:
163  {
164  for ( int i = 0; i < resolution; i++ )
165  {
166  y[ i ] = par1 + par2/(1.0 + par3 * x[ i ]);
167  }
168  break;
169  }
170  case 2:
171  {
172  for ( int i = 0; i < resolution; i++ )
173  {
174  y[ i ] = f_min + (f_max - f_min) * (0.5 * erf((x[i]-par2)/sqrt(2.0*par1)) + 0.5) ;
175  }
176  break;
177  }
178  case 3:
179  {
180  for ( int i = 0; i < resolution; i++ )
181  {
182  y[ i ] = f_max + (f_min - f_max) * (0.5 * erf((x[i]-par2)/sqrt(2.0*par1)) + 0.5);
183  }
184  break;
185  }
186  }
187  plot->btnZoom->setChecked( false );
188  dataPlotClear( data_plot );
189  data_plot->replot();
190  QwtPlotCurve* c1;
191 
192  c1 = us_curve( data_plot, "1DSA Curve" );
193  c1->setStyle ( QwtPlotCurve::Lines );
194  c1->setPen ( QColor( Qt::yellow ) );
195  c1->setSamples( x.data(), y.data(), resolution );
196 
197  data_plot->setAxisAutoScale( QwtPlot::xBottom );
198  data_plot->setAxisAutoScale( QwtPlot::yLeft );
199  data_plot->replot();
200 }
