962 // Generate envelope data
963 int US_vHW_Combine::envel_data( DistrDesc& ddesc )
964 {
965  int nSensit = 50;
966  int nSmooth = 30;
967  int steps;
968  int nepts = 300;
969  int ndpts = ddesc.dsedcs.size();
970  double max_cept = 1.0e-6;
971  double min_cept = 1.0e+6;
972  double sed_bin = ddesc.dsedcs.at( 0 );
973  double his_sum = 0.0;
974  double env_sum = 0.0;
975  double div_scl = (double)nSensit * (double)ndpts * 0.01;
976  double max_step;
977  double sigma;
978  double sed_lo;
979  double sed_hi;
980  double sedc;
981 
982  for ( int jj = 0; jj < ndpts; jj++ )
983  { // get min,max intercept sedimentation coefficients
984  min_cept = qMin( min_cept, ddesc.dsedcs.at( jj ) );
985  max_cept = qMax( max_cept, ddesc.dsedcs.at( jj ) );
986  }
987 
988  // calculate values based on range and sensitivity
989  sed_bin = ( max_cept - min_cept ) / div_scl;
990  max_step = max_cept * 4.0 / 3.0;
991  steps = (int)( max_step / sed_bin );
992 
993  if ( nepts <= steps )
994  { // insure envelope array size bigger than histogram array size
995  nepts = steps + 1;
996  }
997 
998  ddesc.esedcs.fill( 0.0, nepts );
999  ddesc.efreqs.fill( 0.0, nepts );
1000  double bink = 0.0;
1001  double sval = 0.0;
1002  double pisqr = sqrt( M_PI * 2.0 );
1003  double* xval = ddesc.esedcs.data();
1004  double* yval = ddesc.efreqs.data();
1005  double scale = max_step / (double)nepts;
1006 
1007  for ( int jj = 0; jj < nepts; jj++ )
1008  { // initialize envelope values
1009  xval[ jj ] = scale * (double)jj;
1010  yval[ jj ] = 0.0;
1011  }
1012 
1013  sigma = sed_bin * 0.02 * (double)nSmooth;
1014 
1015  for ( int jj = 0; jj < steps; jj++ )
1016  { // calculate histogram values and envelope values based on them
1017  int kbin = 0;
1018  sed_lo = sed_bin * (double)jj;
1019  sed_hi = sed_lo + sed_bin;
1020  sval = ( sed_lo + sed_hi ) * 0.5;
1021 
1022  for ( int kk = 0; kk < ndpts; kk++ )
1023  { // count sedcoeffs within current step range
1024  sedc = ddesc.dsedcs.at( kk );
1025 
1026  if ( sedc >= sed_lo && sedc < sed_hi )
1027  kbin++;
1028  }
1029 
1030  bink = (double)kbin;
1031  his_sum += ( bink * sed_bin ); // bump histogram sum
1032 
1033  if ( kbin > 0 )
1034  { // if non-empty bin, update envelope Y values
1035  for ( int kk = 0; kk < nepts; kk++ )
1036  {
1037  double xdif = ( xval[ kk ] - sval ) / sigma;
1038  yval[ kk ] += ( ( bink / ( sigma * pisqr ) )
1039  * exp( -( xdif * xdif ) / 2.0 ) );
1040  }
1041  }
1042  }
1043 
1044  for ( int kk = 0; kk < nepts; kk++ )
1045  { // accumulate the envelope values sum
1046  env_sum += yval[ kk ];
1047  }
1048 
1049  env_sum *= xval[ 1 ]; // sum times X increment
1050  scale = his_sum / env_sum; // normalizing scale factor
1051 DbgLv(2) << "ED: hsum esum scale " << his_sum << env_sum << scale;
1052 
1053  int fnz = -1;
1054  int lnz = 0;
1055 
1056  for ( int kk = 0; kk < nepts; kk++ )
1057  { // normalize Y values
1058  yval[ kk ] *= scale;
1059 
1060  if ( yval[ kk ] != 0.0 )
1061  {
1062  if ( fnz < 0 ) fnz = kk;
1063  lnz = kk;
1064  }
1065  }
1066 
1067  fnz = qMax( ( fnz - 2 ), 0 );
1068  lnz = qMin( ( lnz + 2 ), nepts );
1069  nepts = 0;
1070 
1071  for ( int ii = fnz; ii < lnz; ii++ )
1072  { // Reduce array to Y's from first to last non-zero
1073  xval[ nepts ] = xval[ ii ];
1074  yval[ nepts ] = yval[ ii ];
1075  nepts++;
1076  }
1077 
1078  ddesc.esedcs.resize( nepts );
1079  ddesc.efreqs.resize( nepts );
1080  ddesc.totconc = 1.0;
1081 
1082  return nepts; // return arrays' size
1083 }
