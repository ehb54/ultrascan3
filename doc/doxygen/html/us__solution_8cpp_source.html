895 // Function to count solution occurrences in protocols
896 int US_Solution::countInProtocols( US_DB2* dbP )
897 {
898  int kntSols = 0;
899 
900  if ( dbP != NULL )
901  { // Get and search XML from each DB protocol record
902  QStringList qry;
903  qry << "get_protocol_desc" << QString::number( US_Settings::us_inv_ID() );
904  dbP->query( qry );
905 
906  if ( dbP->lastErrno() != US_DB2::OK )
907  return 0; // Error exit: unable to read DB record
908 
909  while ( dbP->next() )
910  {
911  QString xmlStr = dbP->value( 3 ).toString();
912  if ( solutionInProtocol( xmlStr, false ) )
913  kntSols++;
914 qDebug() << "kntInPro: DB: kntSols" << kntSols << dbP->value(2).toString();
915  }
916  }
917 
918  else
919  { // Get and search XML from each local protocol file
920  QString datdir = US_Settings::dataDir() + "/projects/";
921  datdir.replace( "\\", "/" ); // Possible Windows fix
922  QStringList rfilt( "R*.xml" ); // "~/ultrascan/data/projects/R*.xml"
923  QStringList pfiles = QDir( datdir ).entryList(
924  rfilt, QDir::Files, QDir::Name );
925  int nfiles = pfiles.count();
926 
927  for ( int ii = 0; ii < nfiles; ii++ )
928  { // Examine each "R000*.xml" file in the directory
929  QString pfpath = datdir + pfiles[ ii ];
930  QFile filei( pfpath );
931  if ( ! filei.open( QIODevice::ReadOnly | QIODevice::Text) )
932  {
933  continue;
934  }
935 
936  QTextStream tsi( &filei );

938  filei.close();

942  }
943  }
944
