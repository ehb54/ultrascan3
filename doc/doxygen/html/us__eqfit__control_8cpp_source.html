751 // Prepare data for plots: get y_delta and data indecies,counts
752 void US_EqFitControl::prepare_data()
753 {
754  double* ydelta = fitpars.y_delta;
755  double* yguess = fitpars.y_guess;
756  double* yraw = fitpars.y_raw;
757  mxspts = 0;
758  ipscnn = (int)ct_plotscn->value();
759  lpscnn = 0;
760  int kpscns = 0;
761  int jpscnn = 0;
762  int dssx = 0;
763  int ntscns = scanfits.size();
764  int niscns = 0;
765  int liscnn = 0;
766  v_dscnx.fill( 0, ntscns );
767  dscnx = v_dscnx.data();
768 
769  for ( int ii = 0; ii < ntscns; ii++ )
770  {
771  EqScanFit* scnf = &scanfits[ ii ];
772  int scann = ii + 1;
773 
774  if ( ! scnf->scanFit ) continue;
775 
776  niscns++; // Bump included-scans count
777  dscnx[ ii ] = dssx; // Save scan's start data index
778  liscnn = scann; // Last included scan number
779  int nspts = scnf->stop_ndx - scnf->start_ndx + 1;
780  mxspts = max( mxspts, nspts ); // Maximum data points in a scan
781 
782  if ( scann >= ipscnn )
783  kpscns++;
784 
785  if ( kpscns == 1 )
786  ipscnn = scann; // Initial plot scan number
787 
788  if ( kpscns > 0 && kpscns < 6 )
789  lpscnn = scann; // Last plot scan number
790 
791  if ( kpscns == 6 )
792  jpscnn = scann; // Next group-of-5 start scan number
793 //qDebug() << "PrD: sn ipsc lpsc jpsc kpsc" << scann << ipscnn << lpscnn
794 // << jpscnn << kpscns;
795 
796  for ( int jj = 0; jj < nspts; jj++ )
797  { // Insure we have freshly calculated deltas for scan data
798  ydelta[ dssx ] = yraw[ dssx ] - yguess[ dssx ];
799  dssx++; // Scan's data index
800  }
801  }
802 
803  plotgrpf = 0;
804  npscns = niscns;
805 
806  if ( rb_pltalld->isChecked() )
807  { // Plot all
808  plotgrpf = 0; // Flag plot-all
809  npscns = niscns; // Plot scans is total included scans
810  lpscnn = liscnn; // Last plot scan is last included
811  jpscnn = 1;
812  }
813 
814  else if ( rb_pltgrp5->isChecked() )
815  { // Plot in group of 5
816  plotgrpf = 5; // Flag plot-5
817  npscns = min( kpscns, 5 ); // Plot scans is at-most 5
818  jpscnn = max( npscns, ( jpscnn - ipscnn ) );
819  }
820 
821  else if ( rb_pltsscn->isChecked() )
822  { // Plot single scan
823  plotgrpf = -1; // Flag plot-1
824  npscns = 1; // Plot scans is one
825  lpscnn = ipscnn; // Last plot scan is same as first
826  jpscnn = 1;
827  }
828 
829  ct_plotscn->setRange( 1, liscnn );
830  ct_plotscn->setSingleStep( jpscnn );
831  ct_plotscn->disconnect();
832  ct_plotscn->setValue( ipscnn );
833  connect( ct_plotscn, SIGNAL( valueChanged( double ) ), SLOT( new_pscan() ) );
834 //qDebug() << "PREP_DATA ipscnn lpscnn liscnn jpscnn" << ipscnn << lpscnn
835 // << liscnn << jpscnn;
836 }
