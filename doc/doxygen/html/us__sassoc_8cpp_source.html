250 double US_Sassoc::monomer_root( double total )
251 {
252  double monomer = total / 2;
253  double diff = monomer / 2;
254  double diff1 = monomer - diff; // = total / 4
255  double test = polynomial( monomer, total );
256 
257  while ( fabs( diff1 ) > 1.0e-15 )
258  {
259  double monomer_old = monomer;
260 
261  if ( test > 0 )
262  monomer = monomer - diff;
263  else
264  monomer = monomer + diff;
265 
266  test = polynomial( monomer, total );
267  diff = fabs( monomer - monomer_old ) / 2;
268  diff1 = monomer_old - monomer;
269  }
270 
271  return monomer;
272 }
