396 // Set simulation parameter as selected in the simparams dialog
397 void US_MwlSpeciesSim::set_parameters( void )
398 {
399  double rad_precis = simparams.radial_resolution * 0.1;
400  simparams.meniscus = qRound( simparams.meniscus
401  / rad_precis ) * rad_precis;
402  simparams.bottom_position = qRound( simparams.bottom_position
403  / rad_precis ) * rad_precis;
404  simparams.bottom = simparams.bottom_position;
405 
406  double delay = simparams.speed_step[ 0 ].delay_hours * 3600.0
407  + simparams.speed_step[ 0 ].delay_minutes * 60.0;
408  double duration = simparams.speed_step[ 0 ].duration_hours * 3600.0
409  + simparams.speed_step[ 0 ].duration_minutes * 60.0;
410  double rspeed = (double)simparams.speed_step[ 0 ].rotorspeed;
411  double scn_rng = (double)( simparams.speed_step[ 0 ].scans - 1 );
412  delay = qRound( delay );
413  duration = qRound( duration );
414  double pi_fac = sq( M_PI / 30.0 );
415  double tim_rng = duration - delay;
416  double tim_inc = tim_rng / scn_rng;
417  double w2t_fac = sq( rspeed ) * pi_fac;
418  double accel = simparams.speed_step[ 0 ].acceleration;
419  double w2t_inc = tim_inc * w2t_fac;
420  double tim_val = 0.0;
421  double w2t_val = 0.0;
422  double aspeed = 0.0;
423  double rpm_inc = rspeed / qCeil( rspeed / ( accel == 0.0 ? 1.0 : accel ) );
424 
425  while ( aspeed < rspeed )
426  { // Walk time and omega2t through acceleration zone
427  aspeed += rpm_inc;
428  tim_val += 1.0;
429  w2t_val += sq( aspeed ) * pi_fac;
430  }
431 
432  while( tim_val < delay )
433  { // Walk time and omega2t up to the first scan
434  tim_val += 1.0;
435  w2t_val += w2t_fac;
436  }
437 
438  simparams.speed_step[ 0 ].time_first = tim_val;
439  simparams.speed_step[ 0 ].w2t_first = w2t_val;
440 
441  while( tim_val < duration )
442  { // Walk time and omega2t up to the last scan
443  tim_val += tim_inc;
444  w2t_val += w2t_inc;
445  }
446 
447  simparams.speed_step[ 0 ].time_last = duration;
448  simparams.speed_step[ 0 ].w2t_last = w2t_val;
449  simparams.speed_step[ 0 ].set_speed = (int)rspeed;
450  simparams.speed_step[ 0 ].avg_speed = rspeed;
451 
452 //*DEBUG*

454 simparams.debug();
