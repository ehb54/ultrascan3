299 // handle any x or y box checked
300 void US_PlotControlPc::checkSet( bool chkd, bool isX, int row )
301 {
302  if ( !chkd ) // only respond if checked
303  return;
304 
305  if ( isX )
306  { // this is a X box
307  for ( int ii = 0; ii < 6; ii++ )
308  { // review all check boxes
309  if ( ii != row )
310  { // in other rows, uncheck X, enable Y
311  xCheck[ ii ]->setChecked( false );
312  yCheck[ ii ]->setEnabled( true );
313  }
314 
315  else
316  { // in this row, disable the Y
317  yCheck[ ii ]->setEnabled( false );
318  }
319  }
320  }
321 
322  else
323  { // this is a Y box
324  for ( int ii = 0; ii < 6; ii++ )
325  { // review all check boxes
326  if ( ii != row )
327  { // in other rows, uncheck Y, enable X
328  yCheck[ ii ]->setChecked( false );
329  xCheck[ ii ]->setEnabled( true );
330  }
331 
332  else
333  { // in this row, disable the X
334  xCheck[ ii ]->setEnabled( false );
335  }
336  }
337  }
338  if ( model->constant_ff0() )
339  { // f/f0 constant, so disable f/f0 X,Y check boxes
340  xCheck[ 4 ]->setEnabled( false );
341  yCheck[ 4 ]->setEnabled( false );
342  xCheck[ 4 ]->setChecked( false );
343  yCheck[ 4 ]->setChecked( false );
344  }
345  if ( model->constant_vbar() )
346  { // Vbar constant, so disable vbar X,Y check boxes
347  xCheck[ 5 ]->setEnabled( false );
348  yCheck[ 5 ]->setEnabled( false );
349  xCheck[ 5 ]->setChecked( false );
350  yCheck[ 5 ]->setChecked( false );
351  }
352 }
