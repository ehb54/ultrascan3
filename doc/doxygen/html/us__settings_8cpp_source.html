515 QString US_Settings::status()
516 {
517  QSettings settings( US3, "UltraScan" );
518  settings.setValue( "status_test", true );
519  settings.sync();
520  settings.remove( "status_test" );
521 
522  switch ( settings.status() ) {
523  case QSettings::NoError :
524  return "";
525  break;
526  case QSettings::AccessError :
527  return QString( "Access error. Check permissions and ownership of %1" ).arg( settings.fileName() );
528  break;
529  case QSettings::FormatError :
530  return QString( "Settings format error. The file %1 is garbled" ).arg( settings.fileName() );
531  break;
532  default:
533  return QString( "Unknown settings error %1. Perhaps remove the file %2 and try again." ).arg( settings.status() ).arg( settings.fileName() );
534  break;
535  }
536 }
