1268  ymax = ct_uplimity->value();

1280  double par2 = mrec.par2;

1322  double yval = ystr + ydif * efac;
1323  isol.x = xval * xscl;
1324  isol.y = yval * yscl;
1325  mrec.isolutes << isol;
1326  xoff += xoinc;
1327  } // END: points-on-curve loop
1328  }
1329 
1330  else if ( ctype == CTYPE_HL )
1331  {
1332  double xval = xmin;
1333  double xinc = xrng / prng;
1334  double yval = end_y;
1335 
1336  for ( int kk = 0; kk < nlpts; kk++ )
1337  { // Loop over points on a line
1338  isol.x = xval * xscl;
1339  isol.y = yval * yscl;
1340  mrec.isolutes << isol;
1341  xval += xinc;

1354  isol.y = yval * yscl;
1355  mrec.isolutes << isol;
