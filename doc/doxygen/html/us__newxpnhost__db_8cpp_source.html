724  if ( le_description->text().isEmpty() || le_serialNumber->text().isEmpty() )

727  tr( "Fill out all fields!"));

732 
733  qDebug() << "Optima Msg Port : " << le_msgPort->text();
734

737 
738 
739  if ( !nonOptima_selected )
740  {
741  if( !rx_desc.exactMatch(le_description->text() ) )
742  {
743  QString mtitle_error = tr( "Error" );
744  QString message_error = tr( "Syntax error for Optima Host Description!\n\nThe description template is the following:\n 'Optima #' (Optima|space|number)" );

749  palette->setColor(QPalette::Base,Qt::white);

751 
752  return;
753  }
754  }
755  else
756  {
757  if( le_description->text().contains("Optima") || le_description->text().contains("optima") )
758  {
759  QString mtitle_error = tr( "Error" );

766  le_description->setPalette(*palette);
767 
768  return;
769  }
770  }
771 
772 
773  // Check for instrument name, port & host duplications:
774  US_Passwd pw;
775  US_DB2* db = use_db ? new US_DB2( pw.getPasswd() ) : NULL;
776  QStringList instrumentNames( "" );
777  QStringList q( "" );
778  QList< int > instrumentIDs;
779  q.clear();
780  q << QString( "get_instrument_names" )
781  << QString::number( 1 ); //ALEXEY '1' for the (only) labID
782 
783  db->query( q );

786  {
787  QString name = db->value( 1 ).toString();
788  instrumentNames << name;
789  int ID = db->value( 0 ).toString().toInt();
790  instrumentIDs << ID;
791  }
792 
793  // Name check
794  qDebug() << "Update Instrument: " << update_instrument;
795  for (QStringList::iterator it = instrumentNames.begin(); it != instrumentNames.end(); ++it)
796  {

812  continue;

815  q << QString( "get_instrument_info_new" )
816  << QString::number( ID );

831  }
