1293 // Calculate and apply an integration to readings data
1294 void US_DataIO::calc_integral( RawData& data, const EditValues& ee )
1295 {
1296  // This function gets a little tricky because we have raw data, but
1297  // want to adjust in the range identified by the user *and*
1298  // not use excluded scans.
1299 
1300  int kk = 0;
1301  QVector< int > included;
1302  QVector< double > integral;
1303  double val_min = 1.e+99;
1304 
1305  for ( int scx = 0; scx < data.scanData.size(); scx++ )
1306  {
1307  if ( ee.excludes.contains( scx ) ) continue;
1308 
1309  included << scx;
1310 
1311  integral << 0.0;
1312 
1313  // Arbitrarily add 1000 fringes to each reading value
1314  // to make sure we don't sum negatives. This is not needed from a
1315  // mathematical point of view, but physically, negative fringes
1316  // do not make sense.
1317  US_DataIO::Scan* sc = &data.scanData[ scx ];
1318 
1319  int r_left = index( data.xvalues, ee.rangeLeft );
1320  int r_right = index( data.xvalues, ee.rangeRight );
1321 
1322  for ( int rr = r_left; rr <= r_right; rr++ )
1323  integral[ kk ] += data.reading( scx, rr ) + 1000.0;
1324 
1325  kk++;
1326 
1327  if ( scx == 0 )
1328  {
1329  for ( int rr = r_left; rr <= r_right; rr++ )
1330  val_min = qMin( val_min, sc->rvalues[ rr ] );
1331  }
1332  }
1333 
1334  // Integral fringe shifts contribute exactly ( r_right - r_left + 1 )
1335  // to the integral, since we use unity stepsize in integral calculation.
1336 
1337  for ( int scx = 1; scx < included.size(); scx++ )
1338  {
1339  US_DataIO::Scan* sc = &data.scanData[ included[ scx ] ];
1340 
1341  int r_left = index( data.xvalues, ee.rangeLeft );
1342  int r_right = index( data.xvalues, ee.rangeRight );
1343  int current = scx;
1344  int previous = scx - 1;
1345  int position = 0;
1346  double points = (double) ( r_right - r_left + 1 );
1347 
1348  while ( integral[ current ] <= integral[ previous ] )
1349  {
1350  integral[ current ] += points;
1351  position++;
1352  }
1353 
1354  while ( integral[ current ] > integral[ previous ] )
1355  {
1356  integral[ current ] -= points;
1357  position--;
1358  }
1359 
1360  // Add the integral steps (which may be negative!) to each datapoint
1361  for ( int rr = r_left; rr <= r_right; rr++ )
1362  sc->rvalues[ rr ] += (double) position;
1363 
1364  double diff1 = integral[ previous ] - integral[ current ];
1365 
1366  double diff2 = integral[ previous ] - ( integral[ current ] + points );

1599  const QVector< Scan >& sl )
1600 {

1602 
1603  QVector< double > cov( order );

1613  // keep track of the total integral at each point.
1614

1616  {

1620  int value_count = sc->rvalues.size();

1623 
1624  // Integrate using trapezoid rule
1625  for ( int jj = 1; jj < value_count; jj++ )
1626  {
1627  double avg = ( sc->rvalues[ jj ] + sc->rvalues[ jj - 1 ] ) / 2.0;
1628  absorbance_integral[ ii ] += avg * delta_r;
1629  }
1630  }
1631 
1632  for ( int ii = 0; ii < scan_count; ii++ )
1633  scan_time[ ii ] = sl[ ii ].seconds;
1634 
1635  US_Matrix::lsfit( coeffs, scan_time, absorbance_integral, scan_count, order );

1638 
1639  for ( int ii = 0; ii < scan_count; ii++ )
1640  {
1641  double fit = 0.0;
1642 
1643  for ( int jj = 0; jj < order; jj++ )
1644  fit += coeffs[ jj ] * pow( sl[ ii ].seconds, jj );
