150 bool US_Model::calc_coefficients( SimulationComponent& component )
151 {
152  // For documentation for the derivation of the coefficient calculations
153  // please see: ../doc/calc_model_coefficient_documentation.pdf
154 
155  bool ok = true;
156  double vbar; // component vbar
157  double volume; // e.g., vbar * mw / AVOGADRO
158  double vol_fac; // volume factor, e.g., 0.75 / M_PI
159  double radius_sphere; // e.g., pow( volume * vol_fac, 1/3 );
160  double rsph_fac; // radius_sphere factor; e.g., 0.06 * PI * VISC
161  double onethird; // one third ( 1.0 / 3.0 )
162  double c; // concentration
163  double s; // sedimentation coefficient
164  double D; // diffusion coefficient
165  double mw; // molecular weight
166  double f; // frictional coefficient
167  double fv; // frictional coefficient (working value)
168  double f_f0; // frictional ratio
169  double f0; // f-zero
170  double s20w;
171  double buoyancyb;
172 // Ideal gas constant times 20 degrees C in Kelvin
173 #define RT (R_GC*K20)
174 
175  // Ensure that we have a vbar we can use
176  vbar = component.vbar20;
177 
178  if ( vbar <= 0.0 )
179  {
180  vbar = TYPICAL_VBAR;
181  component.vbar20 = vbar;
182  }
183 
184  vol_fac = 0.75 / M_PI; // various factors used in calcs below
185  onethird = 1.0 / 3.0;
186  rsph_fac = 0.06 * M_PI * VISC_20W;
187 
188  s = component.s; // component coefficient values
189  D = component.D;
190  mw = component.mw;
191  f = component.f;
192  f_f0 = component.f_f0;
193  c = component.signal_concentration;
194  fv = f;
195 
196  // Adjust buoyancy to be consistent with s-value. Negative s-values require negative buoyancy.
197  // Unless custom grid is used, we set the vbar to the reciprocal of the positive s-value vbar.
198 
199  buoyancyb = 1.0 - ( vbar * DENS_20W );
200 
201  if ( ( s < 0.0 && buoyancyb > 0.0 ) ||
202  ( s > 0.0 && buoyancyb < 0.0 ) )
203  {
204  return false;
205  }
206 
207  // Start with already calculated s if possible
208  if ( s != 0.0 )
209  {
210  s20w = s;
211 
212  // First check s and k (f_f0)
214  if ( f_f0 != 0.0 ) // s and f_f0
215  {
216  // Please see ../doc/calc_model_coefficient_documentation.pdf,
217  // equation 14, with adjustments for units in poise.
218  double numer = 0.02 * vbar * s * VISC_20W * f_f0;
219  f0 = 0.09 * VISC_20W * M_PI * sqrt( numer / buoyancyb );
220  fv = f_f0 * f0;
221  D = RT / ( AVOGADRO * fv );
222  mw = s * RT / ( D * buoyancyb );
223  }
224 
225  // Next check s and D
227  else if ( D != 0.0 ) // s and D
228  {
229  mw = s * RT / ( D * buoyancyb );
230  volume = vbar * mw / AVOGADRO;
231  radius_sphere = pow( volume * vol_fac, onethird );
232  f0 = radius_sphere * rsph_fac;
233  fv = mw * buoyancyb / ( s20w * AVOGADRO );
234  double ff0sv = f_f0;
235  f_f0 = fv / f0;
236  double ffdif = qAbs( ff0sv - f_f0 );
237  f_f0 = ( ffdif < 1.e-5 ) ? ff0sv : f_f0;
238  }
239 
240  // Then check any other s + combinations
242  else if ( mw != 0.0 ) // s and mw
243  {
244  D = s * RT / ( buoyancyb * mw );
245  fv = mw * buoyancyb / ( s20w * AVOGADRO );
246  volume = vbar * mw / AVOGADRO;
247  radius_sphere = pow( volume * vol_fac, onethird );
248  f0 = radius_sphere * rsph_fac;
249  f_f0 = fv / f0;
250  }
252  else if ( f != 0.0 ) // s and f
253  {
254  D = RT / ( AVOGADRO * fv );
255  mw = s * RT / ( D * buoyancyb );
256  volume = vbar * mw / AVOGADRO;
257  radius_sphere = pow( volume * vol_fac, onethird );
258  f0 = radius_sphere * rsph_fac;
259  f_f0 = fv / f0;
260  }
261  //****************************
262  else // do not have 2 valid coeffs
263  ok = false; //****************************
264  }
265 
266  else if ( component.mw != 0.0 )
267  {
268  volume = vbar * mw / AVOGADRO;
269  radius_sphere = pow( volume * vol_fac, onethird );
270  f0 = radius_sphere * rsph_fac;
272  if ( D != 0.0 ) // mw and D
273  {
274  s = D * buoyancyb * mw / RT;
275  fv = mw * buoyancyb / ( s * AVOGADRO );
276  f_f0 = fv / f0;
277  }
279  else if ( f_f0 != 0.0 ) // mw and f_f0
280  {
281  fv = f_f0 * f0;
282  s = mw * buoyancyb / ( AVOGADRO * fv );
283  D = s * RT / ( buoyancyb * mw );
284  }
286  else if ( f != 0.0 ) // mw and f
287  {
288  f_f0 = fv / f0;
289  s = mw * buoyancyb / ( AVOGADRO * fv );
290  D = s * RT / ( buoyancyb * mw );
291  }
292  //****************************
293  else // do not have 2 valid coeffs
294  ok = false; //****************************
295  }
296 
297  else if ( component.D != 0.0 )
298  {
299  if ( f_f0 >= 1.0 ) // D and f_f0
300  {
301  fv = RT / ( AVOGADRO * D );
302  f0 = fv / f_f0;
303  radius_sphere = f0 / ( 0.06 * M_PI * VISC_20W );
304  double volume = ( 4.0 / 3.0 ) * M_PI * pow( radius_sphere, 3.0 );
305  mw = volume * AVOGADRO / vbar;
306  s = mw * buoyancyb / ( AVOGADRO * fv );
307  }
308  //****************************
309  // D and f - not a valid combo
310  //****************************
311  else // do not have 2 valid coeffs
312  ok = false; //****************************
313  }
315  else if ( fv > 0.0 && f_f0 >= 1.0 ) // f and f_f0
316  {
317  f0 = fv / f_f0;
318  D = RT / ( AVOGADRO * fv );
319  radius_sphere = f0 / ( 0.06 * M_PI * VISC_20W );
320  double volume = ( 4.0 / 3.0 ) * M_PI * pow( radius_sphere, 3.0 );
321  mw = volume * AVOGADRO / vbar;
322  s = mw * buoyancyb / ( AVOGADRO * fv );
323  }
324  //****************************
325  else // do not have 2 valid coeffs
326  ok = false; //****************************
327 
328  double df = qAbs( f - fv ) / fv;
329 
330  if ( df > 0.1 )
331  { // Significant change in f: replace and use old as concentration
332 
333  if ( c == 0.0 )
334  component.signal_concentration = f;
335 
336  f = fv;
337  }
338 
339  component.s = s; // set component properties (some re-calculated)
340  component.D = D;
341  component.mw = mw;
342  component.f_f0 = f_f0;
343  component.f = f;
344  component.vbar20 = vbar;
345 
346  return ok;
347 }
