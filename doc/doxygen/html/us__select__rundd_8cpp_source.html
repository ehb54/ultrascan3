304 {

308  QMap< QString, QString > ddmap;

310  QString mdir = US_Settings::dataDir() + "/models";
311  QStringList mfilt( "M*.xml" );

313  .entryList( mfilt, QDir::Files, QDir::Name );
314 
315  for ( int ii = 0; ii < f_names.size(); ii++ )
316  { // Examine each model file
317  QString fname( mdir + "/" + f_names[ ii ] );
318  QFile m_file( fname );
319 
320  if ( ! m_file.open( QIODevice::ReadOnly | QIODevice::Text ) )
321  continue;
322 
323  QXmlStreamReader xml( &m_file );
324 
325  while( ! xml.atEnd() )
326  {
327  xml.readNext();
328 
329  if ( xml.isStartElement() )
330  {
331  if ( xml.name() == "model" )
332  { // Get the description and GUID of model and test it
333  QXmlStreamAttributes attr = xml.attributes();
334  QString mdesc = attr.value( "description" ).toString();
335  QString ddesc = attr.value( "dataDescrip" ).toString();
336  QString mGUID = attr.value( "modelGUID" ).toString();
337  QString eGUID = attr.value( "editGUID" ).toString();
338  int kk = mdesc.lastIndexOf( ".model" );
339  mdesc = ( kk < 1 ) ? mdesc : mdesc.left( kk );
340  QString runid = mdesc.section( ".", 0, -3 );
341  // Skip the model if it has no valid runID part
342  if ( runid.isEmpty() || runid.length() < 2 ) continue;
343 
344  if ( ddesc.isEmpty() )
345  {
346  if ( ddmap.contains( eGUID ) )
347  ddesc = ddmap[ eGUID ];
348  }
349  else
350  {
351  if ( !ddmap.contains( eGUID ) )
352  ddmap[ eGUID ] = ddesc;
353  }
354 
355  // Modify the description field of any MonteCarlo
356  if ( mdesc.contains( "-MC" ) )
357  {
358  kk = mdesc.indexOf( "_mc0" );
359  if ( kk > 0 )
360  {
361  int niters = attr.value( "MCIteration" ).toString()
362  .toInt();
363  mdesc = QString( mdesc ).left( kk ) +
364  QString().sprintf( "_mcN%03d", niters ) +
365  QString( mdesc ).mid( kk + 7 );
366  }
367  }
368 
369  // Save run ID and model string of RunID+GUID+Description
370  QString odesc = runid + "\t" + mGUID + "\t" + mdesc
371  + "\t" + ddesc;
372  mRunIDs << runid;
373  mEdtIDs << eGUID;
374  wDescrs << odesc;
375 if((dbg_level>0) && (!mdesc.contains("-MC_")||mdesc.contains("_mc")))
376  DbgLv(1) << "ScLo: odesc" << odesc;
377  }
378 
379  if ( xml.name() == "analyte" )
380  break;
381  }
382  }
383  }
384 
385  // Do another pass on model descriptions to insure data description
386  // gets used for all models with the same edit GUID
387  for ( int mm = 0; mm < wDescrs.count(); mm++ )
388  {
389  QString mdesc = wDescrs[ mm ];
390  QString ddesc = mdesc.section( "\t", 3, 3 );
391 
392  if ( ddesc.isEmpty() )
393  {
394  QString eGUID = mEdtIDs[ mm ];
395  if ( ddmap.contains( eGUID ) )
396  {
397  ddesc = ddmap[ eGUID ];
398 DbgLv(1) << "ScLo: mm" << mm << "eGUID" << eGUID << "new ddesc" << ddesc;
399  mdesc = mdesc.section( "\t", 0, 2 ) + "\t" + ddesc;
400  wDescrs.replace( mm, mdesc );
401  }
402  }
403  }
404

406 
407  QString rdir = US_Settings::resultDir();

411

418  rlabels.clear();
