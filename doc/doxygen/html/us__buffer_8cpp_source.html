751 void US_Buffer::compositeCoeffs( double* d_coeff, double* v_coeff )
752 {
753  int ncomp = component.size(); // Number of buffer components
754 
755  if ( ncomp == 0 )
756  return; // If 0, nothing to do
757 
758  // Pre-remove water value from components beyond first
759  d_coeff[ 0 ] = (double)( 1 - ncomp ) * DENS_20W;
760  v_coeff[ 0 ] = (double)( 1 - ncomp ) * VISC_20W;
761 
762  for ( int ii = 1; ii < 6; ii++ )
763  { // Initialize other coefficients to zero
764  d_coeff[ ii ] = 0.0;
765  v_coeff[ ii ] = 0.0;
766  }
767 
768  double sumc1 = 0.0;
769  double sumcr = 0.0;
770  double sumc2 = 0.0;
771  double sumc3 = 0.0;
772  double sumc4 = 0.0;
773 
774  for ( int ii = 0; ii < ncomp; ii++ )
775  { // Sum concentration and coefficient values
776  double c1 = concentration[ ii ]; // c ^ 1
777  double cr = sqrt( c1 ); // c ^ 0.5
778  double c2 = c1 * c1; // c ^ 2
779  double c3 = c2 * c1; // c ^ 3
780  double c4 = c3 * c1; // c ^ 4
781  sumc1 += c1; // Sum concentration terms
782  sumcr += cr;
783  sumc2 += c2;
784  sumc3 += c3;
785  sumc4 += c4;
786 
787  // Accumulate weighted coefficient terms
788  d_coeff[ 0 ] += component[ ii ].dens_coeff[ 0 ];
789  d_coeff[ 1 ] += component[ ii ].dens_coeff[ 1 ] * cr;
790  d_coeff[ 2 ] += component[ ii ].dens_coeff[ 2 ] * c1;
791  d_coeff[ 3 ] += component[ ii ].dens_coeff[ 3 ] * c2;
792  d_coeff[ 4 ] += component[ ii ].dens_coeff[ 4 ] * c3;
793  d_coeff[ 5 ] += component[ ii ].dens_coeff[ 5 ] * c4;
794 
795  v_coeff[ 0 ] += component[ ii ].visc_coeff[ 0 ];
796  v_coeff[ 1 ] += component[ ii ].visc_coeff[ 1 ] * cr;
797  v_coeff[ 2 ] += component[ ii ].visc_coeff[ 2 ] * c1;
798  v_coeff[ 3 ] += component[ ii ].visc_coeff[ 3 ] * c2;
799  v_coeff[ 4 ] += component[ ii ].visc_coeff[ 4 ] * c3;
800  v_coeff[ 5 ] += component[ ii ].visc_coeff[ 5 ] * c4;
801  }
802 
803  // Normalize coefficients and multiply by standard powers
804  v_coeff[ 1 ] *= ( 1.0e-3 / sumcr );
805  v_coeff[ 2 ] *= ( 1.0e-2 / sumc1 );
806  v_coeff[ 3 ] *= ( 1.0e-3 / sumc2 );
807  v_coeff[ 4 ] *= ( 1.0e-4 / sumc3 );
808  v_coeff[ 5 ] *= ( 1.0e-6 / sumc4 );
809 
810  d_coeff[ 1 ] *= ( 1.0e-3 / sumcr );
811  d_coeff[ 2 ] *= ( 1.0e-2 / sumc1 );
812  d_coeff[ 3 ] *= ( 1.0e-3 / sumc2 );
813  d_coeff[ 4 ] *= ( 1.0e-4 / sumc3 );
814  d_coeff[ 5 ] *= ( 1.0e-6 / sumc4 );
815 }
