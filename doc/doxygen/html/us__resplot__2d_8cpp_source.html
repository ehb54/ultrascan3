604  rinoi = ri_noise->values[ ii ];
605  if ( do_addrin )
606  rinoi += ri_noise->values[ ii ];

613  if ( do_addtin )
614  tinoi += ti_noise->values[ jj ];
615 
616  vv[ jj ] = edata->value( ii, jj ) - sdata->value( ii, jj )
617  - rinoi - tinoi;

619  kntva++;
620  }
621

623  curv = us_curve( data_plot2, title );
624

627  curv->setSamples( rr, vv, points );
628  }
629
