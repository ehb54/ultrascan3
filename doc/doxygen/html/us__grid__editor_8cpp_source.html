1139 bool US_Grid_Editor::gen_points( double x1, double x2, int np,
1140  bool isLog, bool isMid, QVector<double>& result )
1141 {

1144  result.resize( np );

1146  if ( isLog ) {
1147  if ( x1 < 0 || x2 < 0 ) {
1148  sign = -1;
1149  }
1150  if ( x1 == 0 ) {
1151  x1 = 1e-20;
1152  }
1153  if ( x2 == 0 ) {
1154  x2 = 1e-20;
1155  }
1156  x1 = std::log10( x1 * sign );
1157  x2 = std::log10( x2 * sign );
1158  }
1159  double dx, val;
1160  if ( isMid ) {
1161  dx = ( x2 - x1 ) / static_cast<double>( np );

1168  if ( ii != 0 ) {
1169  val += dx;
1170  }
1171  if ( isLog ) {
1172  result[ii] = sign * std::pow( 10, val );
1173  } else {
1174  result[ii] = val;
1175  }

2367  double Z = std::sqrt( ( 27 * std::pow( Q, 2 ) * 4 * std::pow( Q, 3 ) ) / 108 );
