200  int jend = npoints - 1;

202  double ynorm = qAbs( ys[ 0 ] - ys[ jend ] ); // Y normalizing factor
203  xnorm = ( xnorm == 0.0 ) ? 1.0 : ( 1.0 / xnorm );
204  ynorm = ( ynorm == 0.0 ) ? 1.0 : ( 1.0 / ynorm );
205  double xdif = ( xs[ 0 ] - xgiven ) * xnorm;

208

210  for ( int jj = 0; jj < npoints; jj++ )
211  {
212  xdif = ( xs[ jj ] - xgiven ) * xnorm;
213  ydif = ( ys[ jj ] - ygiven ) * ynorm;
214  double dval = sqrt( sq( xdif ) + sq( ydif ) );
215 
216  if ( dval < dmin )
217  {
218  dmin = dval;
219  jmin = jj;
220  }
221  }
222

224  { // Interpolate a curve point between nearest point and an adjacent one
225  int jmin1 = jmin - 1; // Index to point preceding the nearest
226  int jmin2 = jmin + 1; // Index to point following the nearest
227  double dmin1 = dmin;
228  double dmin2 = dmin;
229

232  xdif = ( xs[ jmin2 ] - xgiven ) * xnorm;

234  dmin2 = sqrt( sq( xdif ) + sq( ydif ) );
235  }

244

247  // determine whether to pick the adjacent point from the one
248  // following or the one preceding. Do this by finding which of the

251  ydif = ( ys[ jmin1 ] - ygiven ) * ynorm;
252  dmin1 = sqrt( sq( xdif ) + sq( ydif ) ); // Dist of preceding
253  xdif = ( xs[ jmin2 ] - xgiven ) * xnorm;
254  ydif = ( ys[ jmin2 ] - ygiven ) * ynorm;
