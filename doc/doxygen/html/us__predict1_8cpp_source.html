303 void US_Predict1::update_plot( void )
304 {
305  double x = 1.1;
306  // From: K.E. van Holde, Biophysical Chemistry, 2nd edition, chapter 4.1
307  // Calculate frictional ratio as a function of axial ratio
308  for ( int i = 0; i < ARRAYSIZE; i++, x += max_x/ARRAYSIZE )
309  {
310  prolate[ i ] = pow( x, -1.0 / 3.0 ) * sqrt( sq( x ) - 1.0 ) /
311  log( x + sqrt( sq( x ) - 1.0 ) );
312 
313  oblate[ i ] = sqrt( sq( x ) - 1.0 ) /
314  ( pow( x, 2.0 / 3.0 ) * atan( sqrt( sq( x ) - 1.0 ) ) );
315 
316  rod[ i ] = pow( 2.0 / 3.0, 1.0 / 3.0 ) * pow( x, 2.0 / 3.0 ) /
317  ( log( 2.0 * x ) - 0.3 );
318 
319  ratio_x[ i ] = x;
320  }
321  prolate_curve->setSamples( ratio_x, prolate, ARRAYSIZE );
322  rod_curve->setSamples( ratio_x, rod, ARRAYSIZE );
323  oblate_curve ->setSamples( ratio_x, oblate, ARRAYSIZE );
324  double maxy = max(prolate_curve->maxYValue(), rod_curve->maxYValue());
325  vline_x[ 0 ] = ratio;
326  vline_x[ 1 ] = ratio;
327  vline_y[ 0 ] = 0;
328  vline_y[ 1 ] = maxy;
329 
330  vline_curve->setPen ( QPen( QBrush( Qt::white ), 1.5 ) );
331  vline_curve->setSamples( vline_x, vline_y, 2 );
332  plot->setAxisScale(QwtPlot::xBottom, 1.0, max_x );
333  plot->setAxisScale(QwtPlot::yLeft, 0.9, maxy );
334  //plot->setAxisAutoScale( QwtPlot::yLeft );
335  plot->setMinimumSize( 650, 350 );

515 void US_Predict1::calc_column( const QString& name, const QString& unit, int column_index, double sphere_val,
516  double prolate_val, double oblate_val, double rod_val, double multiplier)
517 {
518  double min_value = min( min( sphere_val, prolate_val ), min( oblate_val, rod_val ) ) * multiplier;
519  double max_value = max( max( sphere_val, prolate_val ), max( oblate_val, rod_val ) ) * multiplier;
520  int log_min = (int) ( log10( min_value ) ) ;
521  int log_max = (int) ( log10( max_value ) ) ;
522  int power_of_ten = 1e+0;
523  char format_char = 'g';
524  if ( abs( log_min - log_max ) <= 1 )
525  {
526  power_of_ten = log_max;
527  format_char = 'f';
528  }
529  else
530  {
531  power_of_ten = log_min;
532  }
533  power_of_ten *= -1;
534  if ( qFabs(max_value * multiplier * (double)pow(10, power_of_ten)) < 1 )
535  {
536  power_of_ten += (power_of_ten != 0) ? power_of_ten / abs( power_of_ten ) : 1;
537  }
538 
539  titles[ column_index ] ->setText( name + ((unit.isEmpty())?"":" (" + unit + ")" ));
540 
541  if ( power_of_ten != 0 )
542  {
543  titles[ column_index ] ->setText( name + " (" + QString("e%1%2").
544  arg(power_of_ten > 0 ? '+' : '-').arg(qFabs(power_of_ten),2,'f', 0, '0') + ((unit.isEmpty())?"":" " + unit) + ")" );
545  }
546  lb_sphere [ column_index ]->setText( QString::number( sphere_val * multiplier * (double)pow(10, power_of_ten), format_char, 4 ) );
547  lb_prolate[ column_index ]->setText( QString::number( prolate_val * multiplier * (double)pow(10, power_of_ten), format_char, 4 ) );
548  lb_oblate [ column_index ]->setText( QString::number( oblate_val * multiplier * (double)pow(10, power_of_ten), format_char, 4 ) );
549  lb_rod [ column_index ]->setText( QString::number( rod_val * multiplier * (double)pow(10, power_of_ten), format_char, 4 ) );
550 
551 }
