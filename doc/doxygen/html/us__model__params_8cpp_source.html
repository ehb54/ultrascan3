225 // Compute densities for all rows where D2O percent is not zero
226 void US_ModelParams::compute_densities()
227 {
228  const double dnslope = 0.00108766;
229  QList< QObject* > ochilds = this->children();
230  int nzd2pc = 0;
231  int npd2pc = 0;
232  int kk = -1;
233  bool misspc = false;
234  double d2opct = -1.0;
235  double denszpc = 0.0;
236  double density = 0.0;
237  QVector< double > m_dens;
238  QVector< double > m_d2op;
239  QVector< QLineEdit* > m_lned;
240  m_dens.fill( 0.0, nd_orig );
241  m_d2op.fill( 0.0, nd_orig );
242 DbgLv(1) << " cdn: ochilds size" << ochilds.size();
243 
244  for ( int jj = 0; jj < ochilds.size(); jj++ )
245  {
246  QObject* ochild = ochilds[ jj ];
247  QLineEdit* lned = (QLineEdit*) ochild;
248  QString cname = ochild->objectName();
249  QString pname = cname.section( ":", 0, 0 );
250  if ( pname == "D2OP" )
251  { // Handle D2O percent text
252  d2opct = lned->text().toDouble();
253  if ( d2opct == 0.0 )
254  { // If zero, bump count and test if not 1st row
255  nzd2pc++;
256  if ( npd2pc > 0 )
257  misspc = true;
258  }
259  else
260  { // For non-zero, bump count, test if follow zero row
261  npd2pc++;
262  if ( nzd2pc == 0 )
263  misspc = true;
264  }
265  // Save the D2O percent in a work vector
266  kk++;
267  m_d2op[ kk ] = d2opct;
268  }
269  else if ( pname == "DENS" )
270  { // Handle Density text
271  density = lned->text().toDouble();
272  if ( d2opct == 0.0 )
273  { // Save density following percent=0
274  denszpc = density;
275  }
276  else if ( nzd2pc > 0 )
277  { // Compute density where percent!=0; update text box
278  density = denszpc + d2opct * dnslope;
279  lned->setText( QString::number( density ) );
280  }
281  else
282  { // Flag percent>0 density when no z-density yet found
283  density = 0.0;
284  misspc = true;
285  }
286  // Save the density in a work vector; save pointer to textbox
287  m_dens[ kk ] = density;
288  m_lned << lned;
289  }
290  }
291 DbgLv(1) << " cdn: misspc" << misspc;
292 
293  if ( misspc )
294  { // Fill in computed densities before first pct=0
295  for ( int ii = 0; ii < nd_orig; ii++ )
296  {
297  density = m_dens[ ii ];
298  if ( density == 0.0 )
299  { // Compute density when unable to do so during 1st pass
300  d2opct = m_d2op[ ii ];
301  density = denszpc + d2opct * dnslope;
302  m_lned[ ii ]->setText( QString::number( density ) );
303 DbgLv(1) << " cdn: misspc ii" << ii << "density" << density;
304  }
305  }
306  }
307 }
