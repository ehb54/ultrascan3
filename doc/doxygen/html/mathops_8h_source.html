34 #define MULL MULL
35 static av_always_inline av_const int MULL(int a, int b, unsigned shift)
36 {
37  int rt, dummy;
38  if (__builtin_constant_p(shift))
39  __asm__ (
40  "imull %3 \n\t"
41  "shrdl %4, %%edx, %%eax \n\t"
42  :"=a"(rt), "=d"(dummy)
43  :"a"(a), "rm"(b), "i"(shift & 0x1F)
44  );
45  else
46  __asm__ (
47  "imull %3 \n\t"
48  "shrdl %4, %%edx, %%eax \n\t"
49  :"=a"(rt), "=d"(dummy)
50  :"a"(a), "rm"(b), "c"((uint8_t)shift)
51  );
52  return rt;
53 }
