293 bool US_LegacyConverter::sort_files(const QStringList& flist, const QString& path) {
294  QRegularExpression re;
295  re.setPatternOptions(QRegularExpression::CaseInsensitiveOption);
296  QRegularExpressionMatch match;
297  // RunId1991-s0001-c2-s0009-w260-r_-n1.ri2
298  // RunId1991-s0002-c1-s0001-n1.ip1
299  //group1 (.+) = runID
300  //group2 -s(\\d{4,6}) = speed
301  //group3 -c(\\d) = cell number
302  //group4 -s(\\d{4,6}) = scan number
303  //group5 (?:-w(\\d{3})-r_)? = wavelength (optional: some files include it)
304  //group6 -n(\\d) = scan repetition
305  //group7 (ra|ri|ip|fi|wa|wi)\\d = run type
306  QString pattern = "^(.+)-s(\\d{4,6})-c(\\d)-s(\\d{4,6})(?:-w(\\d{3})-r_)?-n(\\d)[.](ra|ri|ip|fi|wa|wi)\\d$";
307  re.setPattern(pattern);
308 
309  QString runid;
310  QMap<QString, QVector<int>> scan_list; // speed-type-cell-wavelength-replica -> scans
311  QMap<QString, QString> fpath_list; // speed-type-cell-wavelength-replica-scan -> filepath
312  QVector<int> speed_list;
313 
314  foreach (QString fpath, flist) {
315  QFileInfo finfo(fpath);
316  QString fname = finfo.fileName();
317  match = re.match(fname.toLower());
318  if (match.hasMatch()) {
319  if (runid.isEmpty()) {
320  runid = match.captured(1);
321  } else {
322  if (QString::compare(runid, match.captured(1)) != 0) {
323  QMessageBox::warning(this, "Error!", "Multiple Run IDs Found!");
324  return false;
325  }
326  }
327  int speed = match.captured(2).toInt();
328  if (! speed_list.contains(speed)) {
329  speed_list << speed;
330  std::sort(speed_list.begin(), speed_list.end());
331  }
332  int speed_id = speed_list.indexOf(speed) + 1;
333  QString cell = match.captured(3);
334  int scan = match.captured(4).toInt();
335  QString wavl = match.captured(5);
336  if (wavl.isEmpty()) {
337  wavl = "000";
338  }
339  QString replica = match.captured(6);
340  QString runtype = match.captured(7);
341  // speed-type-cell-wavelength-replica
342  QString key1 = tr("%1-%2-%3-%4-%5").arg(speed_id).arg(runtype, cell, wavl, replica);
343  if (scan_list.contains(key1)) {
344  scan_list[key1] << scan;
345  } else {
346  QVector<int> ss(1, scan);
347  scan_list.insert(key1, ss);
348  }
349  // speed-type-cell-wavelength-replica-scan
350  QString key2 = tr("%1-%2").arg(key1).arg(scan);
351  if (fpath_list.contains(key2)) {
352  QMessageBox::warning(this, "Error!", tr("Some of scans are redundant!\n\n%1").arg(fname));
353  return false;
354  } else {
355  fpath_list.insert(key2, fpath);
356  }
357  }
358  }
359  QDir dir = QDir(path);
360  QDir subdir = QDir();
361  QMapIterator<QString, QVector<int>> it(scan_list);
362  bool state = true;
363  while (it.hasNext()) {
364  it.next();
365  QString key1 = it.key();
366  dir.mkdir(key1);
367  subdir.setPath(dir.absoluteFilePath(key1));
368  QVector<int> scans = it.value();
369  std::sort(scans.begin(), scans.end());
370  for (int ii = 0; ii < scans.size(); ii++) {
371  QString key2 = tr("%1-%2").arg(key1).arg(scans.at(ii));
372  QString fpath1 = fpath_list.value(key2);
373 
374  QString fname2 = tr("%1").arg(ii + 1).rightJustified(6, '0') + fpath1.right(4);
375  if (! QFile::copy(fpath1, subdir.absoluteFilePath(fname2)) ) {
376  state = false;
377  }
378  }
379  }
380  return state;
381 }
