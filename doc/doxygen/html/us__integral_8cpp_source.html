807 // Sort distribution solute list by s,d values and optionally reduce
808 void US_Integral::sort_distro( QList< S_Solute >& listsols,
809  bool reduce)
810 {
811  int sizi = listsols.size();
812 
813  if ( sizi < 2 )
814  return; // nothing need be done for 1-element list
815 
816  // sort distro solute list depending on selected plot
817  switch (this->plot_x)
818  {
819  case ATTR_D: qSort ( listsols.begin(), listsols.end(), distro_lessthan_d);
820  break;
821  case ATTR_W: qSort( listsols.begin(), listsols.end(), distro_lessthan_w);
822  break;
823  case ATTR_K: qSort( listsols.begin(), listsols.end(), distro_lessthan_k);
824  break;
825  case ATTR_S: qSort( listsols.begin(), listsols.end(), distro_lessthan_s);
826  break;
827  }
828 
829  // check reduce flag
830 
831  if ( reduce )
832  { // skip any duplicates in sorted list
833  S_Solute sol1;
834  S_Solute sol2;
835  QList< S_Solute > reduced;
836  QList< S_Solute >::iterator jj = listsols.begin();
837  sol1 = *jj;
838  reduced.append( *jj ); // output first entry
839  int kdup = 0;
840  int jdup = 0;
841 
842 
843  while ( (++jj) != listsols.end() )
844  { // loop to compare each entry to previous
845  sol2 = *jj; // solute entry
846 
847  if ( sol1.s != sol2.s || sol1.d != sol2.d || sol1.w != sol2.w || sol1.k != sol2.k)
848  { // not a duplicate, so output to temporary list
849  reduced.append( sol2 );
850  jdup = 0;
851  }
852 
853  else
854  { // duplicate, so sum c value;
855  sol2.c += sol1.c; // sum c value
856  sol2.s = ( sol1.s + sol2.s ) * 0.5; // average values
857  sol2.d = ( sol1.d + sol2.d ) * 0.5;
858  sol2.w = ( sol1.w + sol2.w ) * 0.5;
859  sol2.k = ( sol2.k + sol2.k ) * 0.5;
860  reduced.replace( reduced.size() - 1, sol2 );
861  kdup = max( kdup, ++jdup );
862  }
863 
864  sol1 = sol2; // save entry for next iteration
865  }
866 
867  if ( kdup > 0 )
868  { // if some reduction happened, replace list with reduced version
869  double sc = 1.0 / (double)( kdup + 1 );
870 DbgLv(1) << "KDUP" << kdup;
871 //sc = 1.0;
872 
873  for ( int ii = 0; ii < reduced.size(); ii++ )
874  { // first scale c values by reciprocal of maximum replicate count
875  reduced[ ii ].c *= sc;
876  }
877 
878  listsols = reduced;
879 DbgLv(1) << " reduced-size" << reduced.size();
880  }
881  }
882 DbgLv(1) << " sol-size" << listsols.size();
883  return;
884 }
