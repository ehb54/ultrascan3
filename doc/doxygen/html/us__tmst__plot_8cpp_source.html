607 void US_TmstPlot::read_data()
608 {
609  QStringList fkeys;
610  QStringList ffmts;
611  US_TimeState tsobj;
612 DbgLv(1) << "TP:rdda: tmstpath" << tmstpath;
613  QString xdpath = QString( tmstpath ).replace( ".tmst", ".xml" );
614  QString dat_dir = tmstpath.section( "/", 0, -2 ) + "/";
615  QString tsfname = tmstpath.section( "/", -1, -1 );
616  QString runID = tsfname .section( ".", 0, -3 );
617 
618  if ( tsobj.open_read_data( tmstpath ) != 0 )
619  {
620 DbgLv(1) << "TP:rdda: *ERROR* openread :" << tsobj.last_error_message();
621  ndkeys = 0;
622  return;
623  }
624 
625  tsobj.origin( &defvers, &imptype ); // Def version and import type
626  tsobj.field_keys( &fkeys, &ffmts ); // Field keys and formats
627  nfkeys = fkeys.count(); // Field keys count
628  ntimes = tsobj.time_count(); // Time values count
629  ssreso = tsobj.ss_resolution(); // Set speed resolution
630  double tscl = 1.0 / 60.0; // Time scale, seconds-to-minutes
631 DbgLv(1) << "TP:rdda: ntimes nfkeys" << ntimes << nfkeys;
632  ndkeys = nfkeys + 1; // Data keys count (interval added)
633  dvals.fill( QVector< double >(), ndkeys ); // Init data vector of vectors
634 
635  for ( int jt = 0; jt < ntimes; jt++ )
636  { // Build data vectors for each time position
637  tsobj.read_record();
638 
639  for ( int jk = 0; jk < nfkeys; jk++ )
640  { // Set the value for each key
641  QString fkey = fkeys[ jk ];
642  QString ffmt = ffmts[ jk ];
643  // Get value as double, regardless of file format
644  double fval = ffmt.contains( "F" )
645  ? tsobj.time_dvalue( fkey )
646  : (double)tsobj.time_ivalue( fkey );
647  // Convert any time value from seconds to minutes
648  fval *= ( fkey == "Time" ? tscl : 1.0 );
649 //DbgLv(1) << "TP:rdda: jt jk" << jt << jk << "fkey" << fkey << "ffmt" << ffmt
650 // << "fval" << fval;
651 
652  dvals[ jk + 1 ] << fval; // Store value (offset for extra field)
653  }
654  }
655 
656  tsobj.close_read_data();
657 
658  dkeys << "TimeInterval"; // Insert extra delta-t field
659  dkeys << fkeys; // Then insert from file fields
660 
661  double scalea = 1.0 / (double)ntimes; // Averaging scale
662  int tmdx = dkeys.indexOf( "Time" ); // Time data index
663  int scdx = dkeys.indexOf( "Scan" ); // Scan number index
664  int dtdx = dkeys.indexOf( "TimeInterval" ); // Delta-t data index
665 
666  // Build a list of time indexes to scan points
667  for ( int jt = 0; jt < ntimes; jt++ )
668  {
669  int scan = (int)dvals[ scdx ][ jt ];
670  if ( scan > 0 ) scnxs << jt;
671  }
672 
673  nscans = scnxs.count();
674 DbgLv(1) << "TP:rdda: tmdx" << tmdx << "dtdx" << dtdx
675  << "dvals[tmdx]size" << dvals[tmdx].size() << "nscans" << nscans;
676 
677  // Build the vector of TimeInterval-between-scans values
678  int jt1 = scnxs[ 0 ];
679  int jt2 = scnxs[ 1 ];
680  double delta_t = ( dvals[ tmdx ][ jt2 ] - dvals[ tmdx ][ jt1 ] );
681  for ( int jj = 0; jj < jt1; jj++ )
682  dvals[ dtdx ] << delta_t;
683 
684  for ( int ii = 1; ii < scnxs.count(); ii++ )
685  {
686  jt1 = scnxs[ ii - 1 ];
687  jt2 = scnxs[ ii ];
688  delta_t = ( dvals[ tmdx ][ jt2 ] - dvals[ tmdx ][ jt1 ] );
689  for ( int jj = jt1; jj < jt2; jj++ )
690  {
691  dvals[ dtdx ] << delta_t;
692  }
693  }
694 
695  for ( int jj = jt2; jj < ntimes; jj++ )
696  {
697  dvals[ dtdx ] << delta_t;
698  }
699 
700 DbgLv(1) << "TP:rdda: dvals[dtdx]size" << dvals[dtdx].size() << "ntimes" << ntimes;
701 
702  // Now scan values for each key to build minimum,maximum,average
703  for ( int jk = 0; jk < ndkeys; jk++ )
704  {
705  QString dkey = dkeys[ jk ];
706  double dmin = 1e+99;
707  double dmax = -1e+99;
708  double davg = 0.0;
709 
710  for ( int jt = 0; jt < ntimes; jt++ )
711  { // Accumulate min, max, and sum
712  double dval = dvals[ jk ][ jt ];
713  if ( dval == 0.0 ) continue;
714  dmin = qMin( dmin, dval );
715  dmax = qMax( dmax, dval );
716  davg += dval;
717 if(jk==9 && dval<0.0)
718 {
719 int j2=qMax(1,jt);
720 int j1=jt-1;
721 DbgLv(1) << "TP:rdda: DT" << dval << "jt" << jt << "T2,T1"
722  << dvals[1][j2] << dvals[1][j1] << " +++++++";
723 }
724  }
725 
726  davg *= scalea; // Scale sum to get average
727  dmins[ dkey ] = dmin; // Map minimum to key
728  dmaxs[ dkey ] = dmax; // Map maximum to key
729  davgs[ dkey ] = davg; // Map average to key
730 DbgLv(1) << "TP:rdda: key" << dkey << "min,max,avg" << dmin << dmax << davg;
731  }
732 //*DEBUG*
733 for (int jk=0;jk<ndkeys;jk++)
734 {
735  QString dkey=dkeys[jk];
736 DbgLv(1) << "TP:rdda: key" << dkey << " min,max,avg"
737  << dmins[dkey] << dmaxs[dkey] << davgs[dkey];
738 }
739 //*DEBUG*
740 
741  // Build plot options comboBox items
742  pkeys.clear();
743  pkeys << "TimeInterval";
744 
745  for ( int jk = 0; jk < ndkeys; jk++ )
746  {
747  QString dkey = dkeys[ jk ];
748 
749  if ( dkey == "Scan" || dkey == "TimeInterval" )
750  continue;
751 
752  pkeys << dkey;
753  }
754 
755  cb_pltkey->addItems( pkeys );
756 
757  // Populate status information list widget
758  QStringList infotxt;
759  infotxt << tr( "Run ID: %1" ).arg( runID );
760  infotxt << tr( "Directory: %1" ).arg( dat_dir );
761  infotxt << tr( "File: %1" ).arg( tsfname );
762  infotxt << tr( "Time Count: %1" ).arg( ntimes );
763  infotxt << tr( "Scan Count: %1" ).arg( nscans );
764  infotxt << tr( "Field Key Count: %1" ).arg( nfkeys );
765  infotxt << tr( "Data Key Count: %1" ).arg( ndkeys );
766  infotxt << tr( "Set Speed Resolution: %1" ).arg( ssreso );
767 
768  for ( int jk = 0; jk < ndkeys; jk++ )
769  {
770  QString dkey = dkeys[ jk ];
