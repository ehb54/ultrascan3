824 // Sort distribution solute list by s,d values and optionally reduce
825 void US_Density_Match::sort_distro( QList< S_Solute >& listsols,
826  bool reduce )
827 {
828  int sizi = listsols.size();
829 
830  if ( sizi < 2 )
831  return; // nothing need be done for 1-element list
832 
833  // sort distro solute list by s,k values
834 
835  qSort( listsols.begin(), listsols.end(), distro_lessthan );
836 
837  // check reduce flag
838 
839  if ( reduce )
840  { // skip any duplicates in sorted list
841  S_Solute sol1;
842  S_Solute sol2;
843  QList< S_Solute > reduced;
844  QList< S_Solute >::iterator jj = listsols.begin();
845  sol1 = *jj;
846  reduced.append( *jj ); // output first entry
847  int kdup = 0;
848  int jdup = 0;
849 
850  while ( (++jj) != listsols.end() )
851  { // loop to compare each entry to previous
852  sol2 = *jj; // solute entry
853 
854  if ( sol1.s != sol2.s || sol1.d != sol2.d )
855  { // not a duplicate, so output to temporary list
856  reduced.append( sol2 );
857  jdup = 0;
858  }
859 
860  else
861  { // duplicate, so sum c value;
862  sol2.c += sol1.c; // sum c value
863  sol2.s = ( sol1.s + sol2.s ) * 0.5; // average s,d
864  sol2.d = ( sol1.d + sol2.d ) * 0.5;
865  reduced.replace( reduced.size() - 1, sol2 );
866  kdup = max( kdup, ++jdup );
867  }
868 
869  sol1 = sol2; // save entry for next iteration
870  }
871 
872  if ( kdup > 0 )
873  { // if some reduction happened, replace list with reduced version
874  double sc = 1.0 / (double)( kdup + 1 );
875 DbgLv(1) << "KDUP" << kdup;
876 //sc = 1.0;
877 
878  for ( int ii = 0; ii < reduced.size(); ii++ )
879  { // first scale c values by reciprocal of maximum replicate count
880  reduced[ ii ].c *= sc;
881  }
882 
883  listsols = reduced;
884 DbgLv(1) << " reduced-size" << reduced.size();
885  }
886  }
887 DbgLv(1) << " sol-size" << listsols.size();
888  return;
889 }

1072 void US_Density_Match::build_bf_dists()
1073 {

1076  return;

1078  { // Build bfrac distro for each model
1079  build_bf_distro( jj );
1080  }

1082  if ( diff_avg > 0 )

1084  int npoints = alldis[ 0 ].bf_distro.size();
1085  for ( int jj = 0; jj < npoints; jj++ )

1093  dsum += ( alldis[ ii ].bf_distro[ jj ].d * dwt );
1094  wsum += dwt;
1095  }
1096  double dval = dsum / wsum; // average diffusion coefficient
1097  for ( int ii = 0; ii < ndists; ii++ )
