739 // inverse hessian minimization
740 double US_MPI_Analysis::minimize_dmga( DGene& dgene, double fitness )
741 {
742 DbgLv(1) << my_rank << "dg:IHM:minimize dgene comps" << dgene.components.size() << fitness;
743  int vsize = nfloatc;
744  US_Vector vv( vsize ); // Input values
745  US_Vector uu( vsize ); // Vector of derivatives
746  US_Vector zz( vsize ); // Vector of normalizing factors
747 
748  // Create hessian as identity matrix
749  QVector< QVector< double > > hessian( vsize );
750 
751  for ( int ii = 0; ii < vsize; ii++ )
752  {
753  hessian[ ii ] = QVector< double >( vsize, 0.0 );
754  hessian[ ii ][ ii ] = 1.0;
755  }
756 
757  dgmarker.resize( vsize );
758  marker_from_dgene( dgmarker, dgene );
759 
760  // Convert gene to array of normalized doubles and save normalizing factors
761  for ( int ii = 0; ii < vsize; ii++ )
762  {
763  double vval = dgmarker[ ii ];
764  double vpwr = (double)qFloor( log10( vval ) );
765  double vnorm = pow( 10.0, -vpwr );
766  vv.assign( ii, vval * vnorm );
767  zz.assign( ii, vnorm );
768 DbgLv(1) << my_rank << "dg:IHM: ii" << ii << "vval vnorm" << vval << vnorm
769  << "vpwr" << vpwr << "vvi" << vv[ii];
770  }
771 
772  lamm_gsm_df_dmga( vv, uu, zz ); // uu is vector of derivatives
773 
774  static const double epsilon_f = 1.0e-7;
775  static const int max_iterations = 20;
776  int iteration = 0;
777  double epsilon = epsilon_f * fitness * 4.0;
778  bool neg_cnstr = ( vv[ 0 ] < 0.1 ); // Negative constraint?
779 
780  while ( uu.L2norm() >= epsilon_f && iteration < max_iterations )
781  {
782  iteration++;
783  if ( fitness == 0.0 ) break;
784 
785  US_Vector v_s1 = vv;
786  double g_s1 = fitness;
787  double s1 = 0.0;
788  double s2 = 0.5;
789  double s3 = 1.0;
790 DbgLv(1) << my_rank << "dg:IHM: iteration" << iteration << "fitness" << fitness;
791 
792  // v_s2 = vv - uu * s2
793  US_Vector v_s2( vsize );
794  vector_scaled_sum( v_s2, uu, -s2, vv );
795 
796  if ( neg_cnstr && v_s2[ 0 ] < 0.1 )
797  {
798  v_s2.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
799  }
800 
801  double g_s2 = get_fitness_v_dmga( v_s2, zz );
802 DbgLv(1) << my_rank << "dg:IHM: g_s2" << g_s2 << "s2" << s2 << "epsilon" << epsilon;
803 
804  // Cut down until we have a decrease
805  while ( s2 > epsilon && g_s2 > g_s1 )
806  {
807  s3 = s2;
808  s2 *= 0.5;
809  // v_s2 = vv - uu * s2
810  vector_scaled_sum( v_s2, uu, -s2, vv );
811 
812  if ( neg_cnstr && v_s2[ 0 ] < 0.1 )
813  {
814  v_s2.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
815  }
816 
817  g_s2 = get_fitness_v_dmga( v_s2, zz );
818  }
819 DbgLv(1) << my_rank << "dg:IHM: g_s2" << g_s2;
820 
821  // Test for initial decrease
822  if ( s2 <= epsilon || ( s3 - s2 ) < epsilon ) break;
823 
824  US_Vector v_s3( vsize );
825 
826  // v_s3 = vv - uu * s3
827  vector_scaled_sum( v_s3, uu, -s3, vv );
828 
829  if ( neg_cnstr && v_s3[ 0 ] < 0.1 )
830  {
831  v_s3.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
832  }
833 
834  double g_s3 = get_fitness_v_dmga( v_s3, zz );
835 
836  int reps = 0;
837  static const int max_reps = 100;
838 
839  while ( ( ( s2 - s1 ) > epsilon ) &&
840  ( ( s3 - s2 ) > epsilon ) &&
841  ( reps++ < max_reps ) )
842  {
843  double s1_s2 = 1.0 / ( s1 - s2 );
844  double s1_s3 = 1.0 / ( s1 - s3 );
845  double s2_s3 = 1.0 / ( s2 - s3 );
846 
847  double s1_2 = sq( s1 );
848  double s2_2 = sq( s2 );
849  double s3_2 = sq( s3 );
850 
851  double aa = ( ( g_s1 - g_s3 ) * s1_s3 -
852  ( g_s2 - g_s3 ) * s2_s3
853  ) * s1_s2;
854 
855  double bb = ( g_s3 * ( s2_2 - s1_2 ) +
856  g_s2 * ( s1_2 - s3_2 ) +
857  g_s1 * ( s3_2 - s2_2 )
858  ) *
859  s1_s2 * s1_s3 * s2_s3;
860 
861  static const double max_a = 1.0e-25;
862 
863  if ( qAbs( aa ) < max_a )
864  {
865  // Restore gene from array of normalized doubles
866  for ( int ii = 0; ii < vsize; ii++ )
867  {
868  dgmarker[ ii ] = vv[ ii ] / zz[ ii ];
869  }
870 
871  dgene_from_marker( dgmarker, dgene );
872 
873  return fitness;
874  }
875 
876  double xx = -bb / ( 2.0 * aa );
877  double prev_g_s2 = g_s2;
878 
879  if ( xx < s1 )
880  {
881  if ( xx < ( s1 + s1 - s2 ) ) // Keep it close
882  {
883  xx = s1 + s1 - s2; // xx <- s1 + ds
884  if ( xx < 0 ) xx = s1 / 2.0;
885  }
886 
887  if ( xx < 0 ) // Wrong direction!
888  {
889  if ( s1 < 0 ) s1 = 0.0;
890  xx = 0;
891  }
892 
893  // OK, take xx, s1, s2
894  v_s3 = v_s2;
895  g_s3 = g_s2; // 3 <- 2
896  s3 = s2;
897  v_s2 = v_s1;
898  g_s2 = g_s1;
899  s2 = s1; // 2 <- 1
900  s1 = xx; // 1 <- xx
901 
902  // v_s1 = vv - uu * s1
903  vector_scaled_sum( v_s1, uu, -s1, vv );
904 
905  if ( neg_cnstr && v_s1[ 0 ] < 0.1 )
906  {
907  v_s1.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
908  }
909 
910  g_s1 = get_fitness_v_dmga( v_s1, zz );
911  }
912  else if ( xx < s2 ) // Take s1, xx, s2
913  {
914  v_s3 = v_s2;
915  g_s3 = g_s2; // 3 <- 2
916  s3 = s2;
917  s2 = xx; // 2 <- xx
918 
919  // v_s2 = vv - uu * s2
920  vector_scaled_sum( v_s2, uu, -s2, vv );
921 
922  if ( neg_cnstr && v_s2[ 0 ] < 0.1 )
923  {
924  v_s2.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
925  }
926 
927  g_s2 = get_fitness_v_dmga( v_s2, zz );
928  }
929  else if ( xx < s3 ) // Take s2, xx, s3
930  {
931  v_s1 = v_s2;
932  g_s1 = g_s2;
933  s1 = s2; // 2 <- 1
934  s2 = xx; // 2 <- xx
935 
936  // v_s2 = vv - uu * s2
937  vector_scaled_sum( v_s2, uu, -s2, vv );
938 
939  if ( neg_cnstr && v_s2[ 0 ] < 0.1 )
940  {
941  v_s2.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
942  }
943 
944  g_s2 = get_fitness_v_dmga( v_s2, zz );
945  }
946  else // xx >= s3
947  {
948  if ( xx > ( s3 + s3 - s2 ) ) // if xx > s3 + ds/2
949  {
950  // v_s4 = vv - uu * xx
951  US_Vector v_s4( vsize );
952  vector_scaled_sum( v_s4, uu, -xx, vv );
953 
954  if ( neg_cnstr && v_s4[ 0 ] < 0.1 )
955  {
956  v_s4.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
957  }
958 
959  double g_s4 = get_fitness_v_dmga( v_s4, zz );
960 
961  if ( g_s4 > g_s2 && g_s4 > g_s3 && g_s4 > g_s1 )
962  {
963  xx = s3 + s3 - s2; // xx = s3 + ds/2
964  }
965  }
966 
967  // Take s2, s3, xx
968  v_s1 = v_s2;
969  g_s1 = g_s2; // 1 <- 2
970  s1 = s2;
971  v_s2 = v_s3;
972  g_s2 = g_s3;
973  s2 = s3; // 2 <- 3
974  s3 = xx; // 3 <- xx
975 
976  // v_s3 = vv - uu * s3
977  vector_scaled_sum( v_s3, uu, -s3, vv );
978 
979  if ( neg_cnstr && v_s3[ 0 ] < 0.1 )
980  {
981  v_s3.assign( 0, 0.1 + u_random( 100 ) * 0.001 );
982  }
983 
984  g_s3 = get_fitness_v_dmga( v_s3, zz );
985  }
986 
987  if ( qAbs( prev_g_s2 - g_s2 ) < epsilon ) break;
988  } // end of inner loop
989 
990  US_Vector v_p( vsize );
991 
992  if ( g_s2 < g_s3 && g_s2 < g_s1 )
993  {
994  v_p = v_s2;
995  fitness = g_s2;
996  }
997  else if ( g_s1 < g_s3 )
998  {
999  v_p = v_s1;
1000  fitness = g_s1;
1001  }
1002  else
1003  {
1004  v_p = v_s3;
1005  fitness = g_s3;
1006  }
1007 
1008  US_Vector v_g( vsize ); // Vector of derivatives
1009  lamm_gsm_df_dmga( v_p, v_g, zz ); // New gradient in v_g (old in uu)
1010 
1011  US_Vector v_dx( vsize );
1012  // v_dx = v_p - vv
1013  vector_scaled_sum( v_dx, vv, -1.0, v_p );
1014 
1015 
1016  vv = v_p; // vv = v_p
1017 
1018  // dgradient v_dg = v_g - uu
1019  US_Vector v_dg( vsize );
1020  vector_scaled_sum( v_dg, uu, -1.0, v_g );
1021 
1022  US_Vector v_hdg( vsize );
1023 
1024  // v_hdg = hessian * v_dg ( matrix * vector )
1025  for ( int ii = 0; ii < vsize; ii++ )
1026  {
1027  double dotprod = 0.0;
1028 
1029  for ( int jj = 0; jj < vsize; jj++ )
1030  dotprod += ( hessian[ ii ][ jj ] * v_dg[ jj ] );
1031 
1032  v_hdg.assign( ii, dotprod );
1033  }
1034 
1035  double fac = v_dg.dot( v_dx );
1036  double fae = v_dg.dot( v_hdg );
1037  double sumdg = v_dg.dot( v_dg );
1038  double sumxi = v_dx.dot( v_dx );
1039 
1040  if ( fac > sqrt( epsilon * sumdg * sumxi ) )
1041  {
1042  fac = 1.0 / fac;
1043  double fad = 1.0 / fae;
1044 
1045  for ( int ii = 0; ii < vsize; ii++ )
1046  {
1047  v_dg.assign( ii, fac * v_dx[ ii ] - fad * v_hdg[ ii ] );
1048  }
1049 
1050  for ( int ii = 0; ii < vsize; ii++ )
1051  {
1052  for ( int jj = ii; jj < vsize; jj++ )
1053  {
1054  hessian[ ii ][ jj ] +=
1055  fac * v_dx [ ii ] * v_dx [ jj ] -
1056  fad * v_hdg[ ii ] * v_hdg[ jj ] +
1057  fae * v_dg [ ii ] * v_dg [ jj ];
1058 
1059  // It's a symmetrical matrix
1060  hessian[ jj ][ ii ] = hessian[ ii ][ jj ];
1061  }
1062  }
1063  }
1064 
1065  // uu = hessian * v_g ( matrix * vector )
1066  for ( int ii = 0; ii < vsize; ii++ )
1067  {
1068  double dotprod = 0.0;
1069 
1070  for ( int jj = 0; jj < vsize; jj++ )
1071  dotprod += ( hessian[ ii ][ jj ] * v_g[ jj ] );
1072 
1073  uu.assign( ii, dotprod );
1074  }
1075 
1076  } // end while ( uu.L2norm() > epsilon )
1077 
1078  // Restore gene from array of normalized doubles
1079  for ( int ii = 0; ii < vsize; ii++ )
1080  {
1081  dgmarker[ ii ] = vv[ ii ] / zz[ ii ];
1082 DbgLv(1) << my_rank << "dg:IHM: ii" << ii << "vvi zzi dgmi"
1083  << vv[ii] << zz[ii] << dgmarker[ii];
1084  }
1085

1087 DbgLv(1) << my_rank << "dg:IHM: FITNESS" << fitness;
1088
