680 void US_AnalysisBase2::smoothing( double smoothCount )
681 {
682  if ( ! dataLoaded ) return;
683 
684  int smoothPoints = (int) smoothCount;
685 
686  // Restore saved data
687  int index = lw_triples->currentRow();
688  US_DataIO::EditedData* dat = &dataList[ index ];
689 
690  for ( int ii = 0; ii < dat->scanCount(); ii++ )
691  dat->scanData[ ii ].rvalues = savedValues[ ii ];
692 
693  // Smooth the data
694  if ( smoothPoints > 1 )
695  {
696  QVector< double > xwvec( smoothPoints );
697  QVector< double > ywvec( smoothPoints );
698  x_weights = xwvec.data();
699  y_weights = ywvec.data();
700 
701  // Divide the count into 2 standard deviations
702  double increment = 2.0 / smoothCount;
703  int scanPoints = dat->pointCount();
704 
705  // Only calculate half a Gaussian curve, since the other side is symmetric
706  for ( int jj = 0; jj < smoothPoints; jj++ )
707  {
708  x_weights[ jj ] = increment * jj;
709 
710  // Use a standard deviation of 0.7 to narrow the spread and multiply
711  // by 0.7 to scale the result as an empirical weighting factor
712 
713  // Standard deviation = 0.7, mean = 0.0, point = 0.0;
714  y_weights[ jj ] =
715  0.7 * US_Math2::normal_distribution( 0.7, 0.0, x_weights[ jj ] );
716  }
717 
718  // For each scan
719  for ( int ii = 0; ii < dat->scanData.size(); ii++ )
720  {
721  US_DataIO::Scan* scn = &dat->scanData[ ii ];
722 
723  // Loop over all border point centers
724  for ( int jj = 0; jj < smoothPoints; jj++ )
725  {
726  scn->rvalues[ jj ] = smooth_point( ii, jj, -1, smoothPoints );
727  }
728 
729  // Now deal with all non-border points
730  for ( int jj = smoothPoints; jj < scanPoints - smoothPoints - 1; jj++ )
731  {
732  scn->rvalues[ jj ] = smooth_point( ii, jj, 0, smoothPoints );
733  }
734 
735  // Finally the points on the right border
736  for ( int jj = scanPoints - smoothPoints - 1; jj < scanPoints; jj++ )
737  {
738  scn->rvalues[ jj ]
739  = smooth_point( ii, jj, 1, smoothPoints, scanPoints );
740  }
741  } // END: scan loop
742 
743  } // END: smoothPoints > 1
744 
745  data_plot();
746 }

748 double US_AnalysisBase2::smooth_point(
749  int scan, int point, int type, int smoothPoints, int scanPoints )
750 {
751  // type == 0 means no reflection
752  // type == 1 means to reflect on the right
753  // type == -1 means to reflect on the left
754 
755  double sum = 0.0;
756  double sum_y = 0.0;
757  int start;
758  int stop;
759  int direction;
760 
761  // Sum all applicable points left of center
762  if ( type == -1 ) // reflect left
763  {
764  start = point + 1;
765  stop = point + smoothPoints;
766  direction = 1;
767  }
768  else
769  {
770  start = point - 1;
771  stop = point - smoothPoints;
772  direction = -1;
773  }
774 
775  // This is a bit complex because the test for leaving the loop
776  // is different if we are incrementing or decrementing.
777 
778  int position = 0;
779  int k = start;
780 
781  while ( true )
782  {
783  position++;
784  double value = savedValues[ scan ][ k ];
785 
786  if ( type == -1 )
787  {
788  if ( point - position < 0 ) // we need a reflected value
789  {
790  double dy = savedValues[ scan ][ k ] - savedValues[ scan ][ point ];
791  value = savedValues[ scan ][ point ] - dy;
792  }
793  else
794  value = savedValues[ scan ][ point - position ];
795  }
796 
797  sum += value * y_weights[ position ];
798  sum_y += y_weights[ position ];
799 
800  if ( type == -1 )
801  {
802  if ( k > stop ) break;
803  }
804  else
805  {
806  if ( k <= stop ) break;
807  }
808 
809  k += direction;
810  }
811 
812  // Add the center point
813  sum += savedValues[ scan ][ point ] * y_weights[ 0 ];
814  sum_y += y_weights[ 0 ];
815 
816  // Sum all applicable points right of center
817  if ( type == 1 ) // reflect right
818  {
819  start = point - 1;
820  stop = point - smoothPoints;
821  direction = -1;
822  }
823  else
824  {
825  start = point + 1;
826  stop = point + smoothPoints;
827  direction = 1;
828  }
829 
830  position = 0;
831  k = start;
832 
833  while( true )
834  {
835  position++;
836  double value = savedValues[ scan ][ k ];
837 
838  if ( type == 1 )
839  {
840  if ( point + position >= scanPoints ) // Need reflection
841  {
842  double dy = savedValues[ scan ][ k ]
843  - savedValues[ scan ][ point ];
844  value = savedValues[ scan ][ point ] - dy;
845  }
846  else
847  value = savedValues[ scan ][ point + position ];
848  }
849 
850  sum += value * y_weights[ position ];
851  sum_y += y_weights[ position ];
852 
853  if ( type == 1 )
854  {
855  if ( k <= stop ) break;
856  }
857  else
858  {
859  if ( k > stop ) break;
860  }
861 
862  k += direction;
863  }
864 
865  // Normalize by the sum of all weights that were used
866  return sum / sum_y;
867 }
