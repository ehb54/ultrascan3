389 bool US_AssociationsGui::combine_reactants( US_Model::Association* as )
390 {
391  bool combined = false;
392 
393  if ( as->rcomps.size() == 3 && as->stoichs[ 1 ] > 0 )
394  { // 2 reactants and a product: check if reactants are identical
395  int comp1 = as->rcomps[ 0 ];
396  int comp2 = as->rcomps[ 1 ];
397  int stoi1 = as->stoichs[ 0 ];
398  int stoi2 = as->stoichs[ 1 ];
399 
400  combined = ( comp1 == comp2 && stoi1 == stoi2 );
401 
402  if ( combined )
403  { // Identical reactants: combine and double stoichiometry
404  comp2 = as->rcomps [ 2 ]; // Product component
405  stoi2 = as->stoichs[ 2 ]; // Product stoichiometry
406  as->rcomps .resize( 2 ); // Single reactant and product
407  as->stoichs.resize( 2 );
408  as->rcomps [ 0 ] = comp1; // Reactant component
409  as->stoichs[ 0 ] = stoi1 * 2; // Doubled reactant stoich.
410  as->rcomps [ 1 ] = comp2; // Product component
411  as->stoichs[ 1 ] = stoi2; // Product stoichiometry
412  }
413  }
414 
415  return combined;
416 }
