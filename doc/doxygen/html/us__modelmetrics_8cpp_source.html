650 // Sort distribution solute list by s,k values and optionally reduce
651 void US_ModelMetrics::sort_distro( QList< S_Solute >& listsols,
652  bool reduce )
653 {
654  int sizi = listsols.size();
655 
656  if ( sizi < 2 )
657  return; // nothing need be done for 1-element list
658 
659  // sort distro solute list by s,k values
660 
661  qSort( listsols.begin(), listsols.end(), distro_lessthan );
662 
663  // check reduce flag
664 
665  if ( reduce )
666  { // skip any duplicates in sorted list
667  S_Solute sol1;
668  S_Solute sol2;
669  QList< S_Solute > reduced;
670  sol1 = listsols.at( 0 );
671  reduced.append( sol1 ); // output first entry
672  int kdup = 0;
673  int jdup = 0;
674 
675  for ( int jj = 1; jj < sizi; jj++ )
676  { // loop to compare each entry to previous
677  sol2 = listsols.at( jj ); // solute entry
678 
679  if ( sol1.s != sol2.s || sol1.k != sol2.k )
680  { // not a duplicate, so output to temporary list
681  reduced.append( sol2 );
682  jdup = 0;
683  }
684 
685  else
686  { // duplicate: sum c value
687  DbgLv(2) << "DUP: sval svpr jj" << sol1.s << sol2.s << jj;
688  kdup = max( kdup, ++jdup );
689  qreal f = (qreal)( jdup + 1 );
690  sol2.c += sol1.c; // sum c value
691  sol2.s = ( sol1.s * jdup + sol2.s ) / f; // average s,k
692  sol2.k = ( sol1.k * jdup + sol2.k ) / f;
693  reduced.replace( reduced.size() - 1, sol2 );
694  }
695 
696  sol1 = sol2; // save entry for next iteration
697  }
698 
699  if ( reduced.size() < sizi )
700  { // if some reduction happened, replace list with reduced version
701  //double sc = 1.0 / (double)( kdup + 1 );
702 
703  //for ( int ii = 0; ii < reduced.size(); ii++ )
704  // reduced[ ii ].c *= sc;
705 
706  listsols = reduced;
707  }
708  }
709  return;
710 }
