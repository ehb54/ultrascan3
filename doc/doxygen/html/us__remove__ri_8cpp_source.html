1063 double US_RemoveRI::get_rsqrd(double* x, double* y, int np, QVector<double> coeff){
1064  double res = 0;
1065  double ave = 0;
1066  for (int i = 0; i < np; i++){
1067  double yf = 0;
1068  for (int j = 0; j < coeff.size(); j++)
1069  yf += coeff.at(j) * qPow(x[i], j);
1070  res += qPow(y[i] - yf, 2);
1071  ave += y[i];
1072  }
1073  ave /= np;
1074  double var = 0;
1075  for (int i = 0; i < np; i++)
1076  var += qPow(y[i] - ave, 2);
1077  double rs = 1 - (res / var);
1078  return rs;
