894 {
895  DisSys tsys;
896  S_Solute sol_sk;
897  S_Solute sol_xy;
898 
899  model.update_coefficients(); // fill in any missing coefficients
900 
901  QString mdesc = mdescr.section( mdescr.left( 1 ), 1, 1 );
902 
903  // load current colormap
904  tsys.colormap = colormap;
905  tsys.cmapname = cmapname;
906 
907  tsys.run_name = mdesc.section( ".", 0, -3 );
908  QString asys = mdesc.section( ".", -2, -2 );
909  tsys.analys_name = asys.section( "_", 0, -4 ) + "_"
910  + asys.section( "_", -2, -1 );
911  tsys.method = model.typeText();
912  tsys.method = ( mdesc.contains( "-CG" ) )
913  ? tsys.method.replace( "2DSA", "2DSA-CG" )
914  : tsys.method;
915  if ( model.global == US_Model::GLOBAL )
916  {
917  tsys.method = QString( mdesc ).section( '.', 2, 2 )
918  .section( '_', 2, 2 );
919  }
920  tsys.editGUID = model.editGUID;
921  tsys.plot_x = plot_x;
922  tsys.plot_y = plot_y;
923 DbgLv(1) << "LD: method" << tsys.method << "mdesc" << mdesc;
924 
925  if ( tsys.method == "Manual" || tsys.method == "CUSTOMGRID" )
926  {
927  int jj = mdesc.indexOf( ".model" );
928  mdesc = ( jj < 1 ) ? mdesc : mdesc.left( jj );
929  tsys.run_name = mdesc + ".0Z280";
930  tsys.analys_name = "e00_a00_" + tsys.method + "_local";
931 DbgLv(1) << "LD: run_name" << tsys.run_name;
932 DbgLv(1) << "LD: analys_name" << tsys.analys_name;
933  }
934 
935  tsys.distro_type = (int)model.analysis;
936  tsys.monte_carlo = model.monteCarlo;
937 
938  if ( model.monteCarlo )
939  { // Revisit setting if Monte Carlo
940  QString miter = mdescr.section( mdescr.left( 1 ), 6 );
941  int kiter = miter.isEmpty() ? 0 : miter.toInt();
942 
943  if ( kiter < 2 )
944  { // Turn off flag if not composite MC model (is individual MC)
945  tsys.monte_carlo = false;
946  }
947  }
948 
949 
950  te_distr_info->setText( tr( "Run: " ) + tsys.run_name
951  + " (" + tsys.method + ")\n " + tsys.analys_name );
952  plt_zmin_co = 1e+8;
953  plt_zmax_co = -1e+8;
954  plt_zmin_zp = 100.0;
955  plt_zmax_zp = 0.0;
956  int nsolmc = model.components.size();
957 
958  // read in and set distribution s,k,c,... values
959  if ( tsys.distro_type != (int)US_Model::COFS )
960  {
961  double tot_conc = 0.0;
962 
963  for ( int jj = 0; jj < nsolmc; jj++ )
964  {
965  sol_sk.s = model.components[ jj ].s * 1.0e13;
966  sol_sk.k = model.components[ jj ].f_f0;
967  sol_sk.c = model.components[ jj ].signal_concentration;
968  sol_sk.w = model.components[ jj ].mw;
969  sol_sk.v = model.components[ jj ].vbar20;
970  sol_sk.d = model.components[ jj ].D * 1.0e7;
971  sol_sk.f = model.components[ jj ].f;
972 
973  sol_xy = sol_sk;
974  sol_xy.s = ( plot_x == ATTR_S ) ? sol_sk.s : sol_xy.s;
975  sol_xy.s = ( plot_x == ATTR_K ) ? sol_sk.k : sol_xy.s;
976  sol_xy.s = ( plot_x == ATTR_W ) ? sol_sk.w : sol_xy.s;
977  sol_xy.s = ( plot_x == ATTR_V ) ? sol_sk.v : sol_xy.s;
978  sol_xy.s = ( plot_x == ATTR_D ) ? sol_sk.d : sol_xy.s;
979  sol_xy.s = ( plot_x == ATTR_F ) ? sol_sk.f : sol_xy.s;
980  sol_xy.k = ( plot_y == ATTR_S ) ? sol_sk.s : sol_xy.k;
981  sol_xy.k = ( plot_y == ATTR_K ) ? sol_sk.k : sol_xy.k;
982  sol_xy.k = ( plot_y == ATTR_W ) ? sol_sk.w : sol_xy.k;
983  sol_xy.k = ( plot_y == ATTR_V ) ? sol_sk.v : sol_xy.k;
984  sol_xy.k = ( plot_y == ATTR_D ) ? sol_sk.d : sol_xy.k;
985  sol_xy.k = ( plot_y == ATTR_F ) ? sol_sk.f : sol_xy.k;
986 
987  tsys.sk_distro << sol_sk;
988  tsys.xy_distro << sol_xy;
989 
990  plt_zmin_co = qMin( plt_zmin_co, sol_sk.c );
991  plt_zmax_co = qMax( plt_zmax_co, sol_sk.c );
992  tot_conc += sol_sk.c;
993  }
994 DbgLv(1) << "LD: zmin zmax totconc" << plt_zmin_co << plt_zmax_co << tot_conc;
995 
996  // sort and reduce distributions
997  sort_distro( tsys.sk_distro, false );
998  sort_distro( tsys.xy_distro, true );
999  int nsolsk = tsys.sk_distro.size();
1000  int nsolxy = tsys.xy_distro.size();
1001 DbgLv(1) << "LD: nsolsk nsolxy nsolmc" << nsolsk << nsolxy << nsolmc;
1002  tsys.sk_distro_zp.clear();
1003  tsys.xy_distro_zp.clear();
1004 
1005  // Create Z-as-percentage version of distributions
1006 
1007  for ( int jj = 0; jj < nsolmc; jj++ )
1008  {
1009  double cozpc;
1010 
1011  if ( jj < nsolsk )
1012  {
1013  sol_sk = tsys.sk_distro[ jj ];
1014  cozpc = sol_sk.c * 100.0 / tot_conc;
1015  sol_sk.c = cozpc;
1016  plt_zmin_zp = qMin( plt_zmin_zp, cozpc );
1017  plt_zmax_zp = qMax( plt_zmax_zp, cozpc );
1018  tsys.sk_distro_zp << sol_sk;
1019  }
1020 
1021  if ( jj < nsolxy )
1022  {
1023  sol_xy = tsys.xy_distro[ jj ];

1230 // Sort distribution solute list by s,k values and optionally reduce
1231 void US_Pseudo3D_Combine::sort_distro( QList< S_Solute >& listsols,
1232  bool reduce )
1233 {
1234  int sizi = listsols.size();
1235 
1236  if ( sizi < 2 )
1237  return; // nothing need be done for 1-element list
1238 
1239  // sort distro solute list by s,k values
1240 
1241  qSort( listsols.begin(), listsols.end(), distro_lessthan );
1242 
1243  // check reduce flag
1244 
1245  if ( reduce )
1246  { // skip any duplicates in sorted list
1247  S_Solute sol1;
1248  S_Solute sol2;
1249  QList< S_Solute > reduced;
1250  QList< S_Solute >::iterator jj = listsols.begin();
1251  sol1 = *jj;
1252  reduced.append( *jj ); // output first entry
1253  int kdup = 0;

1255

1257  { // loop to compare each entry to previous
1258  sol2 = *jj; // solute entry
1259 
1260  if ( sol1.s != sol2.s || sol1.k != sol2.k )
1261  { // not a duplicate, so output to temporary list
1262  reduced.append( sol2 );
1263  jdup = 0;
1264  }
1265 
1266  else
1267  { // duplicate, so sum c value;
1268  sol2.c += sol1.c; // sum c value
1269  sol2.s = ( sol1.s + sol2.s ) * 0.5; // average s,k
1270  sol2.k = ( sol1.k + sol2.k ) * 0.5;
1271  reduced.replace( reduced.size() - 1, sol2 );
1272  kdup = max( kdup, ++jdup );
1273  }
1274 
1275  sol1 = sol2; // save entry for next iteration
1276  }
1277 
1278  if ( kdup > 0 )
1279  { // if some reduction happened, replace list with reduced version
1280  double sc = 1.0 / (double)( kdup + 1 );
1281 DbgLv(1) << "KDUP" << kdup;
1282 //sc = 1.0;
1283 
1284  for ( int ii = 0; ii < reduced.size(); ii++ )
1285  { // first scale c values by reciprocal of maximum replicate count
1286  reduced[ ii ].c *= sc;
1287  }
1288 
1289  listsols = reduced;
1290 DbgLv(1) << " reduced-size" << reduced.size();
1291  }
1292  }
