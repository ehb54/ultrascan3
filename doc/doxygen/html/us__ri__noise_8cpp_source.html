85  double* coeffs = new double[ order ];

89 
90  // Calculate the integral of each scan which is needed for the least-squares
91  // polynomial fit to correct for radially invariant baseline noise. We also
92  // keep track of the total integral at each point.
93 
94  int scan = 0;
95 
96  for ( int i = 0; i < scan_count; i++ )
97  {
98  if ( ! includes.contains( i ) ) continue;
99 
100  const US_DataIO::Scan* s = &data.scanData[ i ];
101 
102  int indexLeft = US_DataIO::index( data.xvalues, range_left );
103  int indexRight = US_DataIO::index( data.xvalues, range_right );
104 
105  double delta_r = ( range_right - range_left ) /
106  ( indexRight - indexLeft );
107 
108  absorbance_integral[ scan ] = 0.0;
109 
110  // Integrate using trapezoid rule
111  for ( int j = indexLeft + 1; j <= indexRight; j++ )
112  {
113  double avg = ( s->rvalues[ j ] + s->rvalues[ j - 1 ] ) / 2.0;
114 
115  absorbance_integral[ scan ] += avg * delta_r;
116  }
117 
118  scan++;
119  }
120 
121  scan = 0;
122 
123  for ( int i = 0; i < scan_count; i++ )
124  {
125  if ( ! includes.contains( i ) ) continue;
126  scan_time[ scan++ ] = data.scanData[ i ].seconds;
127  }
128 
129  US_Matrix::lsfit( coeffs, scan_time, absorbance_integral, scan, order );
130 
131  residuals.clear();
132  scan = 0;
133 
134  for ( int i = 0; i < scan_count; i++ )
135  {
136  if ( includes.contains( i ) )
137  {
138  fit[ scan ] = 0;
139 
140  for ( int j = 0; j < order; j++ )
141  fit[ scan ] += coeffs[ j ] * pow( data.scanData[ i ].seconds, j );
142 
143  residuals << absorbance_integral[ scan ] - fit[ scan ];
144  scan++;
145  }
146  else
147  residuals << 0.0; // Fill in holes for deleted scans
148  }
