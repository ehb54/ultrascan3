485 // generate histogram data
486 int US_DistribPlot::histo_data( QVector< double >& xvec,
487  QVector< double >& yvec )
488 {
489  int steps;
490  int stoff = 0;
491  double max_cept = 1.0e-6;
492  double min_cept = 1.0e+6;
493  double sed_bin = dsedcs.at( 0 );
494  double div_scl = (double)nSensit * (double)divsCount * 0.01;
495  double max_step;
496  double sed_lo;
497  double sed_hi;
498  double sedc;
499 
500  for ( int jj = 0; jj < divsCount; jj++ )
501  { // get min,max intercept sedimentation coefficients
502  max_cept = max( max_cept, dsedcs.at( jj ) );
503  min_cept = min( min_cept, dsedcs.at( jj ) );
504  }
505 
506  // calculate values based on range and sensitivity
507  sed_bin = ( max_cept - min_cept ) / div_scl;
508  max_step = max_cept * 4.0 / 3.0;
509  steps = (int)( max_step / sed_bin );
510  stoff = (int)( min_cept / sed_bin ) - 1;
511  stoff = qMax( stoff, 0 );
512  steps -= stoff;
513 
514  xvec.fill( 0.0, steps );
515  yvec.fill( 0.0, steps );
516  double* sval = xvec.data(); // sedcoeff array
517  double* bink = yvec.data(); // bin count array
518 
519  for ( int jj = 0; jj < steps; jj++ )
520  { // accumulate histogram values
521  int kbin = 0;
522  sed_lo = sed_bin * (double)( jj + stoff ); // low bin sedcoeff
523  sed_hi = sed_lo + sed_bin; // high bin sedcoeff
524  sval[ jj ] = ( sed_lo + sed_hi ) * 0.5; // mid bin sedcoeff
525 
526  for ( int kk = 0; kk < divsCount; kk++ )
527  { // accumulate count of sedcoeff values in current bin
528  sedc = dsedcs.at( kk );
529 
530  if ( sedc >= sed_lo && sedc < sed_hi )
531  kbin++;
532  }
533 
534  bink[ jj ] = (double)kbin; // current bin count
535  }
536 
537  return steps; // return arrays' size
538 }
