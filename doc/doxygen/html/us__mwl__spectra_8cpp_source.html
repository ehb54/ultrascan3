1478 // Utility to find an index in a QVector<double> to a value epsilon match
1479 int US_MwlSpectra::dvec_index( QVector< double >& dvec, const double dval )
1480 {
1481  int indx = dvec.indexOf( dval ); // Try to find an exact match
1482 
1483  if ( indx < 0 )
1484  { // If no exact match was found, look for a match within epsilon
1485  double dmin = dval;
1486 
1487  for ( int jj = 0; jj < dvec.size(); jj++ )
1488  { // Search doubles vector
1489  if ( dvirt_equal( dvec[ jj ], dval ) )
1490  { // If vector value matches within epsilon, break and return
1491  indx = jj;
1492  break;
1493  }
