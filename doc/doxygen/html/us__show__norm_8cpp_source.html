636 void US_show_norm::set_limits()
637 {
638  double smin = 1.0e30;
639  double smax = -1.0e30;

641  double kmax = -1.0e30;

648  data_plot->setAxisTitle( QwtPlot::yLeft, ya_title );
649

651  return;
652 
653  // find min,max for X,Y distributions
654  for ( int ii = 0; ii < model->components.size(); ii++ )
655  {
656  double sval = model->components[ii].s;
657  double kval = model->components[ii].f_f0;
658  smin = qMin( smin, sval );
659  smax = qMax( smax, sval );
660  kmin = qMin( kmin, kval );
661  kmax = qMax( kmax, kval );
662  }
663 
664  // adjust minima, maxima
665  sinc = ( smax - smin ) / 10.0;
666  kinc = ( kmax - kmin ) / 10.0;
667  sinc = ( sinc <= 0.0 ) ? ( smin * 0.05 ) : sinc;
668  kinc = ( kinc <= 0.0 ) ? ( kmin * 0.05 ) : kinc;
669  DbgLv(1) << "SL: real smin smax kmin kmax" << smin << smax << kmin << kmax;
670  smin -= sinc;
671  smax += sinc;
672  kmin -= kinc;
673  kmax += kinc;
674  DbgLv(1) << "SL: adjusted smin smax kmin kmax" << smin << smax << kmin << kmax;
675 
676  if ( auto_sxy )
677  { // Set auto limits on X and Y
678  sinc = pow( 10.0, qFloor( log10( smax ) ) - 3.0 );
679  kinc = pow( 10.0, qFloor( log10( kmax ) ) - 3.0 );
680  if ( qAbs( ( smax - smin ) / smax ) < 0.001 )
681  { // Put padding around virtually constant value
682  smin -= sinc;
683  smax += sinc;
684  }
685  if ( qAbs( ( kmax - kmin ) / kmax ) < 0.001 )
686  { // Put padding around virtually constant value
687  kmin -= kinc;
688  kmax += kinc;
689  }
690  // Make sure limits are nearest reasonable values
691  smin = qFloor( smin / sinc ) * sinc;
692  smax = qFloor( smax / sinc ) * sinc + sinc;
693  smin = ( plot_x != ATTR_S ) ? qMax( smin, 0.0 ) : smin;
694  smin = ( plot_x == ATTR_K ) ? qMax( smin, 0.5 ) : smin;
695  kmin = qFloor( kmin / kinc ) * kinc;
696  kmax = qFloor( kmax / kinc ) * kinc + kinc;
697 
698  DbgLv(1) << "SL: setVal kmin kmax" << kmin << kmax;
699  ct_plt_smin->setValue( smin );
700  ct_plt_smax->setValue( smax );
701  ct_plt_kmin->setValue( kmin );
702  ct_plt_kmax->setValue( kmax );
703 
704  plt_smin = smin;
705  plt_smax = smax;
706  plt_kmin = kmin;
707  plt_kmax = kmax;
708  }
709  else
710  {
711  plt_smin = ct_plt_smin->value();
712  plt_smax = ct_plt_smax->value();

721 // Sort distribution solute list by s,k values and optionally reduce
722 void US_show_norm::sort_distro( QList< S_Solute >& listsols,
723  bool reduce )
724 {
725  int sizi = listsols.size();
726 
727  if ( sizi < 2 )
728  return; // nothing need be done for 1-element list
729 
730  // sort distro solute list by s,k values
731 
732  qSort( listsols.begin(), listsols.end(), distro_lessthan );
733 
734  // check reduce flag
735 
736  if ( reduce )
737  { // skip any duplicates in sorted list
738  S_Solute sol1;
739  S_Solute sol2;
740  QList< S_Solute > reduced;
741  QList< S_Solute >::iterator jj = listsols.begin();
742  sol1 = *jj;
743  reduced.append( *jj ); // output first entry
744  int kdup = 0;
745  int jdup = 0;
746 
747  while ( (++jj) != listsols.end() )
748  { // loop to compare each entry to previous
749  sol2 = *jj; // solute entry
750 
751  if ( sol1.s != sol2.s || sol1.k != sol2.k )
752  { // not a duplicate, so output to temporary list
753  reduced.append( sol2 );
754  jdup = 0;
755  }
756 
757  else
758  { // duplicate, so sum c value;
759  sol2.c += sol1.c; // sum c value
760  sol2.s = ( sol1.s + sol2.s ) * 0.5; // average s,k
761  sol2.k = ( sol1.k + sol2.k ) * 0.5;
762  reduced.replace( reduced.size() - 1, sol2 );
763  kdup = max( kdup, ++jdup );
764  }
765 
766  sol1 = sol2; // save entry for next iteration
767  }
768 
769  if ( kdup > 0 )
770  { // if some reduction happened, replace list with reduced version
771  double sc = 1.0 / (double)( kdup + 1 );
772 DbgLv(1) << "KDUP" << kdup;
773 //sc = 1.0;
774 
775  for ( int ii = 0; ii < reduced.size(); ii++ )
776  { // first scale c values by reciprocal of maximum replicate count
777  reduced[ ii ].c *= sc;
778  }
779 
780  listsols = reduced;
781 DbgLv(1) << " reduced-size" << reduced.size();
782  }
783  }
784 DbgLv(1) << " sol-size" << listsols.size();
785  return;
786 }
