417 bool US_SimParamsGui::load_params( const QString& load_init, US_SimulationParameters& params ) {
418  bool loaded = false;
419 
420  // Save rotor parameters as they are upon entry
421  QString sv_rcal = params.rotorCalID;
422  double sv_rcoef1 = params.rotorcoeffs[ 0 ];
423  double sv_rcoef2 = params.rotorcoeffs[ 1 ];
424 
425  if ( params.load_simparms( load_init ) == 0 )
426  {
427  // Override loaded rotor parameters with the settings at entry
428  params.rotorCalID = sv_rcal;
429  params.rotorcoeffs[ 0 ] = sv_rcoef1;
430  params.rotorcoeffs[ 1 ] = sv_rcoef2;
431  loaded = true;
432  for ( int jj = 0; jj < params.speed_step.count(); jj++ )
433  {
434  US_SimulationParameters::SpeedProfile* ss = &params.speed_step[ jj ];
435  ss->avg_speed = (double)ss->rotorspeed;
436  ss->set_speed = ss->rotorspeed;
437  }
438  simparams = params;
439  }
440  return loaded;
441 }

635 void US_SimParamsGui::check_delay( void )
636 {
637  QVector< int > hours;
638  QVector< double > minutes;
639  QVector< int > speed;
640 
641  speed.clear();
642  speed << 0;
643 
644  int steps = simparams.speed_step.size();
645 
646  for ( int i = 0; i < steps; i++ )
647  {
648  hours << 0;
649  minutes << 0.0;
650 
651  US_SimulationParameters::SpeedProfile* ss = &simparams.speed_step[ i ];
652  speed << ss->rotorspeed;
653 
654  int lower_limit = 1 +
655  ( abs( (speed[ i + 1 ] - speed[ i ] ) ) + 1 ) / ss->acceleration;
656 
657  hours [ i ] = lower_limit / 3600;
658  int secs = lower_limit - hours[ i ] * 3600;
659  minutes[ i ] = (double)secs / 60.0;
660  }
661 
662  //cnt_delay_mins ->setMinimum( minutes[ current_speed_step ] );
663  cnt_delay_hours->setMinimum( hours [ current_speed_step ] );
664  double mindmin = minutes[ current_speed_step ] + ( 1.0 / 30.0 );
665 
666  US_SimulationParameters::SpeedProfile* sp = &simparams.speed_step[ current_speed_step ];
667 
668  if ( sp->delay_hours == hours[ current_speed_step] &&
669  sp->delay_minutes < mindmin )
670  {
671  sp->delay_minutes = mindmin;
672  cnt_delay_mins->setValue( mindmin );
673  }
674 
675  if ( sp->delay_hours < (int)hours[ current_speed_step ] )
676  {
677  sp->delay_hours = hours[ current_speed_step ];
678  cnt_delay_hours->setValue( hours[ current_speed_step ] );
679  }
680 
681  if ( sp->duration_hours == 0 && sp->duration_minutes < sp->delay_minutes + 1 )
682  {
683  sp->duration_minutes = (int) sp->delay_minutes + 1;
684 
685  cnt_duration_mins->setValue( sp->duration_minutes);
686 
687  cnt_duration_mins->setMinimum( sp->delay_minutes + 1 );
688  }
689  else if ( sp->duration_hours > 0)
690  {
691  cnt_duration_mins->setMinimum( 0 );
692  }
693  else if ( sp->duration_hours == 0 &&
694  sp->duration_minutes > sp->delay_minutes + 1 )
695  {
696  cnt_duration_mins->disconnect();
697  cnt_duration_mins->setMinimum( sp->delay_minutes + 1 );
698  connect( cnt_duration_mins, SIGNAL( valueChanged ( double ) ),
699  SLOT ( update_duration_mins( double ) ) );
700  }
701 }

821  fn = fdir + "sp_" + fnam;
822  }
823 
824  QFile f( fn );
825

827  {
828  if( QMessageBox::No == QMessageBox::warning( this,

954 {
955  simparams.meshType = (US_SimulationParameters::MeshType)mesh;
956 
957  // By default, the simpoints can be set by the user
958  cnt_simpoints->setEnabled( true );
959 
960  if ( mesh == 3 )
961  {
962  QMessageBox::information( this,
963  tr( "UltraScan Information" ),
964  tr( "Please note:\n\n"
965  "The radial mesh file should have\n"
966  "the following format:\n\n"
967  "radius_value1\n"
968  "radius_value2\n"
969  "radius_value3\n"
970  "etc...\n\n"
971  "Radius values smaller than the meniscus or\n"
972  "larger than the bottom of the cell will be\n"
973  "excluded from the concentration vector." ) );
974 
975  QFile meshfile( US_Settings::appBaseDir() + "/mesh.dat");
976

978  {

980  simparams.mesh_radius.clear();
981 
982  bool first = true;
983 
984  while ( ! ts.atEnd() )
985  {
986  double value;
987  ts >> value;
988 
989  // Ignore values outside the meniscus/bottom range
990  if ( value >= simparams.meniscus && value <= simparams.bottom )
991  {
992  if ( first )
993  {
994  if ( value > simparams.meniscus )
995  {
996  simparams.mesh_radius << simparams.meniscus;
997  }
998 
999  first = false;
1000  }
1001 
1002  simparams.mesh_radius << value;
1003  }
1004  }
1005 
1006  meshfile.close();
1007 
1008  int mesh_size = simparams.mesh_radius.size();
1009 
1010  if ( simparams.mesh_radius[ mesh_size - 1 ] < simparams.bottom )
1011  {
1012  simparams.mesh_radius << simparams.bottom;
1013  }
1014 
1015 
1016  simparams.simpoints = simparams.mesh_radius.size();
1017  cnt_simpoints->setValue( (double) simparams.simpoints );
1018 
1019  // Can't change the simulation points after defining a mesh
1020  cnt_simpoints->setEnabled( false );
1021  }
1022  else
1023  {
1024  //simparams.mesh = 0; // Set to default mesh
1025  cmb_mesh->setCurrentIndex( 0 );
1026 
1027  // By default, the simpoints can be set by the user
1028  cnt_simpoints->setEnabled( true );
1029 
1030  QMessageBox::warning( this,
1031  tr( "UltraScan Warning" ),
1032  tr( "Please note:\n\n"
1033  "UltraScan could not open the mesh file!\n"
1034  "The file:\n\n" ) +
1035  US_Settings::appBaseDir() + tr( "/mesh.dat\n\n"
1036  "could not be opened." ) );
1037  }
1038  }
1039  report_mods();
