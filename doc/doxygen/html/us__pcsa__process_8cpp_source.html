741 
742  if ( abort )
743  return;
744 
745  kctask++; // Bump count of completed subgrid tasks

772 
773  }
774  }
775 
776  else

778 
779  qSort( mrecs ); // Sort model records by variance
780

782  << "mrec0 taskx,vari" << mrecs[0].taskx << mrecs[0].variance;
