335 // Public function to calculate dimension value types and data ranges
336 void US_Plot3D::calculateAxes( )
337 {
338  const double VROUND = 10.0;
339  const double VNEGOF = (1.0/VROUND);
340  const double MAX_ANNO = (99.9/VROUND);
341  int powrz;
342  double xval;
343  double yval;
344  double zval;
345  double xround = VROUND;
346  double yround = VROUND;
347  double xavg;
348  double yavg;
349  // If there is only one data point, ensure that the minimum and maximum of each axis is not the same
350  if ( data_points == 1 )
351  {
352  zmin *= 0.90;
353  zmax *= 1.10;
354  ymin *= 0.90;
355  ymax *= 1.10;
356  xmin *= 0.90;
357  xmax *= 1.10;
358  }
359 
360  // extend x,y,z ranges a bit
361  xval = ( xmax - xmin ) * 0.05;
362  xmin -= xval;
363  xmax += xval;
364  yval = ( ymax - ymin ) * 0.05;
365  ymin -= yval;
366  ymax += yval;
367 
368  // determine a normalizing power-of-ten for x, y and z
369  if ( model != nullptr )
370  {
371  xavg = ( qAbs( xmin ) + qAbs( xmax ) ) * 0.5;
372  yavg = ( qAbs( ymin ) + qAbs( ymax ) ) * 0.5;
373  x_norm = MAX_ANNO / xavg;
374  y_norm = MAX_ANNO / yavg;
375  }
376  z_norm = MAX_ANNO / zmax;
377 
378  powrx = qRound( log10( x_norm ) );
379  powry = qRound( log10( y_norm ) );
380  powrz = qRound( log10( z_norm ) );
381 
382  if ( xyzdat != nullptr )
383  {
384  powrx = ( powrx > (-2) && powrz < 6 ) ? 0 : powrx;
385  powry = ( powry > (-2) && powry < 6 ) ? 0 : powry;
386  }
387 
388  x_norm = pow( 10.0, (double)powrx );
389  y_norm = pow( 10.0, (double)powry );
390  z_norm = pow( 10.0, (double)powrz );
391  x_norm *= 0.1;
392 
393  if ( model != nullptr && ( xavg * x_norm ) > MAX_ANNO )
394  {
395  x_norm *= 0.1;
396  powrx--;
397  }
398 
399  if ( model != nullptr && ( yavg * y_norm ) > MAX_ANNO )
400  {
401  y_norm *= 0.1;
402  powry--;
403  }
404 
405  if ( ( zmax * z_norm ) > MAX_ANNO )
406  {
407  z_norm *= 0.1;
408  powrz--;
409  }
410  // If vbar is selected, change the scale to be between 0.1 and 1 instead of 1 and 10
411  if ( xatitle.startsWith("vbar", Qt::CaseInsensitive) )
412  {
413  x_norm /= 10.0;
414  xround *= 10.0;
415  }
416  if ( yatitle.startsWith("vbar", Qt::CaseInsensitive) )
417  {
418  y_norm /= 10.0;
419  yround *= 10.0;
420  }
421  // Apply the norm to the ranges
422  xmax *= x_norm;
423  ymax *= y_norm;
424  xmin *= x_norm;
425  ymin *= y_norm;
426  zmin *= z_norm;
427  zmax *= z_norm;
428 
429  // Round the ranges to the nearest power of 10 if a model is plotted
430  if ( model != nullptr )
431  {
432  xmin = (double)( (int)( xmin * xround ) ) / xround;
433  xmax = (double)( (int)( xmax * xround ) + 1 ) / xround;
434  xmin = ( xmin < 0.0 ) ? ( xmin - VNEGOF ) : xmin;
435  ymin = (double)( (int)( ymin * yround ) ) / yround;
436  ymax = (double)( (int)( ymax * yround ) + 1 ) / yround;
437  ymin = ( ymin < 0.0 ) ? ( ymin - VNEGOF ) : ymin;
438  zmax = (double)( (int)( zmax * VROUND ) + 1 ) / VROUND;
439  zmin = 0.0;
440  }
441  DbgLv(2) << "P3D:cA: powrx powry powrz" << powrx << powry << powrz;
442  DbgLv(2) << "P3D:cA: x_norm y_norm z_norm" << x_norm << y_norm << z_norm;
443  DbgLv(2) << "P3D:cA: xmin xmax ymin ymax zmin zmax" << xmin << xmax << ymin << ymax << zmin << zmax;
444 }
