811 // Reset for a refinement iteration
812 void US_MPI_Analysis::iterate( void )
813 {
814  // Just return if the number of iterations exceeds the max
815  // or if the last two iterations converged and are essentially identical
816  if ( ++iterations > max_iterations )
817  {
818  qDebug() << "++ Maximum refinement iterations has been reached:" << max_iterations;
819  return;
820  }
821 
822  double diff = simulation_values.variance - previous_values.variance;
823  double adiff = qAbs( diff );
824  bool ssame = false;
825  int hiters = max_iterations / 2;
826 
827  if ( iterations > 2 )
828  {
829  if ( diff > 0.0 && iterations > hiters )
830  { // Variance is getting worse
831  iterations -= 2;
832  simulation_values = previous_values;
833  simulation_values.solutes = previous_values.solutes;
834  simulation_values.variances = previous_values.variances;
835  simulation_values.ti_noise = previous_values.ti_noise;
836  simulation_values.ri_noise = previous_values.ri_noise;
837  qDebug() << "++ Refinement iteration degrading" << diff
838  << ". Reverting to iteration" << iterations;
839  return;
840  }
841 
842  if ( adiff < min_variance_improvement )
843  {
844  qDebug() << "++ Refinement iteration improvement near-zero:" << adiff;
845  return;
846  }
847 
848  int nsols = previous_values.solutes.size();
849 
850  if ( nsols == simulation_values.solutes.size() )
851  {
852  ssame = true;
853 
854  for ( int jj = 0; jj < nsols; jj++ )
855  {
856  if ( previous_values.solutes[ jj ] != simulation_values.solutes[ jj ] )
857  { // Mismatch: may need to iterate
858  ssame = false;
859  break;
860  }
861 
862  else
863  { // Test if concentrations are almost the same
864  double pconc = previous_values.solutes[ jj ].c;
865  double cconc = simulation_values.solutes[ jj ].c;
866  double cdiff = qAbs( ( pconc - cconc ) / pconc );
867 
868  if ( cdiff > 1.0e-1 ) // Greater than 10% concentration diff?
869  { // Significant concentration mismatch: may need to iterate
870  ssame = false;
871  break;
872  }
873  }
874  }
875  }
876  }
877 
878  if ( ssame )
879  { // Solutes same as previous: no more iterations
880  qDebug() << "++ Refinement iteration solutes same as previous:"
881  << previous_values.solutes.size()
882  << simulation_values.solutes.size();
883  return;
884  }
885 
886  // Save the most recent simulation values for the next time
887  previous_values.variance = simulation_values.variance;
888  previous_values.solutes = simulation_values.solutes;
889  previous_values.variances = simulation_values.variances;
890  previous_values.ti_noise = simulation_values.ti_noise;
891  previous_values.ri_noise = simulation_values.ri_noise;
892 
893  // Set up for another round at depth 0
894  Sa_Job job;
895  job.mpi_job.dataset_offset = current_dataset;
896  job.mpi_job.dataset_count = datasets_to_process;
897  job.mpi_job.meniscus_value = meniscus_value;
898  job.mpi_job.bottom_value = bottom_value;
899  max_experiment_size = min_experiment_size;
900 
901  QVector< US_Solute > prev_solutes = simulation_values.solutes;
902 
903  for ( int i = 0; i < orig_solutes.size(); i++ )
904  {
905  job.solutes = orig_solutes[ i ];
906 
907  // Add back all non-zero Solutes to each job
908  // Ensure there are no duplicates
909  for ( int s = 0; s < prev_solutes.size(); s++ )
910  {
911  if ( ! job.solutes.contains( prev_solutes[ s ] ) )
912  {
913  job.solutes << prev_solutes[ s ];
914  }
915  }
916 
917  job_queue << job;
918 
919  // Bump max solutes per subgrid to new observed max
920  max_experiment_size = qMax( max_experiment_size,
921  job.solutes.size() );
922  }
923 
924  // Clear depth and calculated-solutes lists
925  worker_depth.fill( 0 );
926  max_depth = 0;
927  for ( int ii = 0; ii < calculated_solutes.size(); ii++ )
928  calculated_solutes[ ii ].clear();
929 
930  return;
931 }
