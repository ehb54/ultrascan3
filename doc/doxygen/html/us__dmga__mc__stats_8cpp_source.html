337  conf99lo = vmean - 2.576 * sdevi;
338  conf99hi = vmean + 2.576 * sdevi;
339  conf95lo = vmean - 1.960 * sdevi;
340  conf95hi = vmean + 1.960 * sdevi;

344  clim99hi = mode_cen - conf99lo;
345  }
346

348  stats[ 1 ] = vhi; // Minimum
349  stats[ 2 ] = vmean; // Mean

357  stats[ 10 ] = conf95hi; // 95% Confidence Interval High
358  stats[ 11 ] = conf99lo; // 99% Confidence Interval Low
359  stats[ 12 ] = conf99hi; // 99% Confidence Interval High
360  stats[ 13 ] = sdevi; // Standard Deviation
361  stats[ 14 ] = sderr; // Standard Error
362  stats[ 15 ] = vari; // Variance
363  stats[ 16 ] = corr; // Correlation Coefficient
364  stats[ 17 ] = binsz; // Number of Bins
365  stats[ 18 ] = area; // Distribution Area
366  stats[ 19 ] = clim95lo; // 95% Confidence Limit Low
367  stats[ 20 ] = clim95hi; // 95% Confidence Limit High
368  stats[ 21 ] = clim99lo; // 99% Confidence Limit Low
369  stats[ 22 ] = clim99hi; // 99% Confidence Limit High
370 
371  return is_fixed;
