257 void US_FDS_FileManager::parse_files( void )
258 {
259  cb_triple->disconnect();
260  cb_triple->clear();
261  scaninfo.clear();
262  QString line, str, str1, str2;
263  QStringList tokens, triplelist;
264  int i;
265  triplelist.clear();
266  for ( i=0; i<files.size(); i++)
267  {
268  tmp_scaninfo.filename = files.at(i);
269  if ( files.at(i).contains( "A", Qt::CaseInsensitive ) )
270  {
271  tmp_scaninfo.channel = "A";
272  }
273  else
274  {
275  tmp_scaninfo.channel = "B";
276  }
277  QFile f(source_dir + tmp_scaninfo.filename);
278  if ( f.open( QFile::ReadOnly ) )
279  {
280  QTextStream ts(&f);
281  line = ts.readLine();
282  tokens = line.split(" ", QString::SkipEmptyParts);
283  tmp_scaninfo.date = tokens.at(0);
284  tmp_scaninfo.time = tokens.at(1) + " " + tokens.at(2);
285  tmp_scaninfo.voltage = tokens.at(4).toInt();
286  tmp_scaninfo.gain = tokens.at(6).toInt();
287  tmp_scaninfo.range = tokens.at(8).toInt();
288  tmp_scaninfo.include = true;
289  tokens.clear();
290  line = ts.readLine();
291  tokens = line.split(" ", QString::SkipEmptyParts);
292  tmp_scaninfo.cell = tokens.at(1).toInt();
293  tmp_scaninfo.rpm = tokens.at(3).toInt();
294  tmp_scaninfo.seconds = tokens.at(4).toDouble();
295  tmp_scaninfo.omega_s = tokens.at(4).toDouble();
296  tmp_scaninfo.lambda = tokens.at(6).toInt();
297  tmp_scaninfo.triple = str1.setNum(tmp_scaninfo.cell) + " / " +
298  tmp_scaninfo.channel + " / " +
299  str2.setNum(tmp_scaninfo.lambda);
300  tmp_scaninfo.gainset = "V: " + str1.setNum(tmp_scaninfo.voltage) + ", G: " +
301  str.setNum(tmp_scaninfo.gain) + ", R: " +
302  str2.setNum(tmp_scaninfo.range);
303  tmp_scaninfo.include = true;
304  tmp_scaninfo.x.clear();
305  tmp_scaninfo.y.clear();
306  while (!ts.atEnd())
307  {
308  line = ts.readLine();
309  tokens.clear();
310  tokens = line.split(" ", QString::SkipEmptyParts);
311  tmp_scaninfo.x.append(tokens.at(0).toDouble() );
312  tmp_scaninfo.y.append(tokens.at(1).toDouble() );
313  }
314  }
315  triplelist.append(tmp_scaninfo.triple);
316  progress->setValue( 100 * ( i+1.0 )/files.size() );
317  scaninfo.append(tmp_scaninfo);
318  }
319  if (scaninfo.size() == 0)
320  {
321  QMessageBox::warning (this, tr( "Attention:" ),
322  tr( "The selected directory does not contain fluorescence files...\n"
323  "Please try again." ),
324  QMessageBox::Ok );
325  return;
326  }
327  triplelist.removeDuplicates();
328  triplelist.sort();
329  cb_triple->addItems( triplelist );
330  connect( cb_triple, SIGNAL( currentIndexChanged( int ) ),
