544  int ncolors = 1;
545  for ( int ii = 1; ii <= nsteps; ii++ )
546  {

548  ncolors += npoints;
549  }
550

552  qreal radi = (qreal)heightp / 2.0; // radius is half of height
553  qreal thikc = radi / (qreal)ncolors; // thickness is radius/#colors
554  qreal x1leg = heightp + thikc + 4.0; // rectangle x start
555  qreal x2leg = (qreal)widthp - 4.0; // rectangle x end
556  qreal y1leg = (qreal)heightp; // rectangle y start
557  qreal dyleg = y1leg / (qreal)ncolors; // y increment legend rect
558  qreal thikl = dyleg + 1.0; // thickness legend band lines
559  y1leg -= ( dyleg / 2.0 ); // adjust start legend y
560  radi -= ( thikc / 2.0 ); // back off initial radius a bit

563 
564  // loop through steps determining color range in each
565

571

574  qreal delr = (qreal)( ec.red() - cc.red() ) / rngp;
575  qreal delg = (qreal)( ec.green() - cc.green() ) / rngp;
576  qreal delb = (qreal)( ec.blue() - cc.blue() ) / rngp;
577

580  qreal dvlg = (qreal)cc.green();
581  qreal dvlb = (qreal)cc.blue();

584

607  }
