512 bool US_Database::parse_database_url( const QString& database_url)
513 {
514  if ( database_url.isEmpty() || !database_url.startsWith( "mysql://" ))
515  {
516  return false;
517  }
518  // regular expression for "mysql://{$username}:{$password}@{$host}/{$dbname}|{$_SESSION['email']}"
519  QRegularExpression rx;
520  rx.setPattern( R"(mysql:\/\/([^:]+):([^@]*)@([^\/]+)\/(\w+)\|([\w@.\-\+]+))" );
521  QRegularExpressionMatch match = rx.match( database_url );
522  if ( !match.hasMatch() )
523  {
524  return false;
525  }
526  QString username = match.captured( 1 ).trimmed();
527  QString password = match.captured( 2 ).trimmed();
528  QString host = match.captured( 3 ).trimmed();
529  QString dbname = match.captured( 4 ).trimmed();
530  QString email = match.captured( 5 ).trimmed();
531  if ( le_description->text().trimmed().isEmpty() && !host.isEmpty() && !dbname.isEmpty() )
532  {
533  le_description->setText( host + " - " + dbname );
534  }
535  if ( le_password->text().trimmed().isEmpty() && !password.isEmpty() )
536  {
537  le_password->setText( password );
538  }
539  le_username->setText( username );
540  le_dbname ->setText( dbname );
541  le_host ->setText( host );
542  le_investigator_email->setText( email );
543  // validate input fields
544  values_updated();
545  return true;
546 }
